1//! # Antigravity Session Holder and PTY Spawning
2//!
3//! This module manages active interactive sessions with the Google Antigravity CLI.
4//! It handles non-echoing PTY spawning, non-blocking asynchronous output draining,
5//! and automatic resource cleanup to prevent zombie processes.
6
7//! 
8//! ## Search Tags
9//! #session

11use std::collections::HashMap;
12use std::path::Path;
13use std::time::Instant;
14use tokio::sync::Mutex;
15pub use super::pty_spawner::spawn_session;
16use super::pty_spawner::SessionHolder;

18#[derive(Clone, Debug, serde::Serialize, serde::Deserialize)]
19pub struct AskState {
20    pub msg_id: i32,
21    pub questions: Vec<crate::cli::AskQuestionData>,
22    pub current_index: usize,
23    pub answers: Vec<String>,
24    pub current_bitmap: String,
25    pub waiting_for_write_in: bool,
26}
27
28pub struct SessionManager {
29    pub(crate) holders: Mutex<HashMap<String, SessionHolder>>,
30    pub(crate) running_runs: Mutex<std::collections::HashSet<String>>,
31    pub(crate) active_asks: Mutex<HashMap<String, AskState>>,
32}
33
34impl Default for SessionManager {
35    fn default() -> Self {
36        Self {
37            holders: Mutex::new(HashMap::new()),
38            running_runs: Mutex::new(std::collections::HashSet::new()),
39            active_asks: Mutex::new(HashMap::new()),
40        }
41    }
42}
43
44fn terminate_duplicates(
45    sid: &str,
46    cid: Option<i64>,
47    tid: Option<i64>,
48    holders: &mut HashMap<String, SessionHolder>,
49) {
50    if let Some(c) = cid {
51        holders.retain(|id, h| id == sid || h.chat_id != Some(c) || h.topic_id != tid);
52    }
53}
54
55impl SessionManager {
56    pub fn new() -> Self {
57        Self::default()
58    }
59
60    pub async fn active_count(&self) -> usize {
61        let holders = self.holders.lock().await;
62        holders.len()
63    }
64
65    pub async fn is_running(&self, session_id: &str) -> bool {
66        let runs = self.running_runs.lock().await;
67        runs.contains(session_id)
68    }
69
70    pub async fn set_running(&self, session_id: &str, running: bool) {
71        let mut runs = self.running_runs.lock().await;
72        if running {
73            runs.insert(session_id.to_string());
74        } else {
75            runs.remove(session_id);
76        }
77    }
78
79    pub async fn is_ask_active(&self, session_id: &str) -> bool {
80        let asks = self.active_asks.lock().await;
81        asks.contains_key(session_id)
82    }
83
84    pub async fn set_ask_active(&self, session_id: &str, active: bool) {
85        let mut asks = self.active_asks.lock().await;
86        if active {
87            asks.entry(session_id.to_string()).or_insert_with(|| AskState {
88                msg_id: 0,
89                questions: Vec::new(),
90                current_index: 0,
91                answers: Vec::new(),
92                current_bitmap: String::new(),
93                waiting_for_write_in: false,
94            });
95        } else {
96            asks.remove(session_id);
97        }
98    }
99
100    pub async fn set_ask_state(&self, session_id: &str, state: AskState) {
101        let mut asks = self.active_asks.lock().await;
102        asks.insert(session_id.to_string(), state);
103    }
104
105    pub async fn get_ask_state(&self, session_id: &str) -> Option<AskState> {
106        let asks = self.active_asks.lock().await;
107        asks.get(session_id).cloned()
108    }
109
110    pub async fn cleanup_expired(&self) {
111        let now = Instant::now();
112        self.holders.lock().await.retain(|_, h| {
113            h.child.try_wait().map(|s| s.is_none()).unwrap_or(false)
114                && now.duration_since(h.last_active) < std::time::Duration::from_secs(86400)
115        });
116    }
117
118    pub async fn ensure_session(
119        &self,
120        session_id: &str,
121        workspace: &Path,
122        cmd_name: &str,
123        cmd_args: &[String],
124        env: &HashMap<String, String>,
125    ) -> Result<(), String> {
126        self.cleanup_expired().await;
127
128        let chat_id = env.get("TUNER_CHAT_ID").and_then(|s| s.parse::<i64>().ok());
129        let topic_id = env.get("TUNER_TOPIC_ID").and_then(|s| s.parse::<i64>().ok());
130
131        let mut holders = self.holders.lock().await;
132        let is_running = holders.get_mut(session_id)
133            .map(|h| h.child.try_wait().map(|s| s.is_none()).unwrap_or(false))
134            .unwrap_or(false);
135
136        if is_running {
137            terminate_duplicates(session_id, chat_id, topic_id, &mut holders);
138            if let Some(h) = holders.get_mut(session_id) { h.last_active = Instant::now(); }
139            return Ok(());
140        }
141        holders.remove(session_id);
142
143        terminate_duplicates(session_id, chat_id, topic_id, &mut holders);
144
145        let holder = spawn_session(workspace, cmd_name, cmd_args, env)?;
146        holders.insert(session_id.to_string(), holder);
147        Ok(())
148    }
149
150    pub async fn terminate(&self, session_id: &str) -> bool {
151        self.holders.lock().await.remove(session_id).is_some()
152    }
153
154    pub async fn abort(&self, chat_id: i64, topic_id: Option<i64>) -> usize {
155        let mut h = self.holders.lock().await;
156        let prev = h.len();
157        h.retain(|_, val| val.chat_id != Some(chat_id) || val.topic_id != topic_id);
158        prev - h.len()
159    }
160
161    pub async fn terminate_all(&self) {
162        self.holders.lock().await.clear();
163    }
164
165    pub async fn write_to_session(&self, session_id: &str, input: &str) -> Result<bool, String> {
166        let holders = self.holders.lock().await;
167        if let Some(holder) = holders.get(session_id) {
168            if input.starts_with('\x1b') {
169                holder.write_input(input)?;
170            } else {
171                for c in input.chars() {
172                    let s = c.to_string();
173                    holder.write_input(&s)?;
174                    let delay = if cfg!(test) {
175                        tokio::time::Duration::from_millis(1)
176                    } else {
177                        tokio::time::Duration::from_millis(100)
178                    };
179                    tokio::time::sleep(delay).await;
180                }
181            }
182            Ok(true)
183        } else {
184            Ok(false)
185        }
186    }
187
188    pub async fn is_active(&self, session_id: &str) -> bool {
189        let mut h = self.holders.lock().await;
190        if let Some(holder) = h.get_mut(session_id) {
191            if holder.child.try_wait().map(|s| s.is_none()).unwrap_or(false) { return true; }
192            h.remove(session_id);
193        }
194        false
195    }
196}
