22    pub(crate) fn build_env(&self) -> HashMap<String, String> {
23        let mut env: HashMap<_, _> = std::env::vars().filter(|(k, _)| k != "CODEX_SANDBOX_NETWORK_DISABLED").collect();
24        let mut add = |k: &str, v: String| env.insert(k.into(), v);
25        add("TUNER_AGENT_NAME", "main".into());
26        add("TUNER_CHAT_ID", self.config.chat_id.to_string());
27        if let Some(tid) = self.config.topic_id { add("TUNER_TOPIC_ID", tid.to_string()); }
28        add("TUNER_TRANSPORT", self.config.transport.clone());
29        let home = std::env::var("HOME").unwrap_or_else(|_| ".".to_string());
30        let tuner_home = std::path::PathBuf::from(&home).join(".tuner");
31        let shared_memory_path = tuner_home.join("SHAREDMEMORY.md");
32        add("TUNER_HOME", tuner_home.to_string_lossy().into());
33        add("TUNER_SHARED_MEMORY_PATH", shared_memory_path.to_string_lossy().into());
34        env
35    }

37    async fn ensure_interactive_session(&self, session_id: &str, _workspace: &Path, env: &HashMap<String, String>) -> Result<(), String> {
38        let agy_ws = self.agy_workspace();
39        let mut args = vec!["--add-dir".into(), agy_ws.to_string_lossy().into(), "--conversation".into(), session_id.into()];
40        if self.config.permission_mode == "bypassPermissions" { args.push("--dangerously-skip-permissions".into()); }
41        args.extend(vec!["--prompt-interactive".into(), "".into()]);
42        self.sessions.ensure_session(session_id, &agy_ws, "agy", &args, env).await
43    }

45    async fn run_in_pty_session(
46        &self,
47        session_id: &str,
48        prompt: &str,
49        env: &HashMap<String, String>,
50        agy_ws: &Path,
51    ) -> Result<(String, String, std::process::ExitStatus), String> {
52        let sessions = self.sessions.clone();
53        let sid_str = session_id.to_string();
54        sessions.set_running(&sid_str, true).await;
55
56        let res = async {
57            self.ensure_interactive_session(session_id, agy_ws, env).await?;
58
59            let needs_init = self.sessions.holders.lock().await
60                .get_mut(session_id)
61                .map(|h| { let u = !h.initialized; h.initialized = true; u })
62                .unwrap_or(false);
63
64            if needs_init {
65                wait_for_pty_prompt(&self.sessions, session_id).await?;
66                tokio::time::sleep(tokio::time::Duration::from_millis(200)).await;
67            }
68
69            let p = events::agy_state_root(Some(env)).join("brain").join(session_id)
70                .join(".system_generated/logs/transcript_full.jsonl");
71            let sz = std::fs::metadata(&p).map(|m| m.len()).unwrap_or(0);
72
73            let input_prompt = format!("{}\r", prompt);
74            self.sessions.write_to_session(session_id, &input_prompt).await?;
75
76            super::polling::wait_for_log_completion(&self.sessions, session_id, Some(p), sz).await?;
77
78            use std::os::unix::process::ExitStatusExt;
79            Ok((String::new(), String::new(), std::process::ExitStatus::from_raw(0)))
80        }.await;
81
82        sessions.set_running(&sid_str, false).await;
83        if !sessions.is_active(&sid_str).await {
84            sessions.set_ask_active(&sid_str, false).await;
85        }
86        res
87    }

89    async fn run_oneshot(
90        &self,
91        cmd_args: &[String],
92        env: &HashMap<String, String>,
93        workspace: &Path,
94    ) -> Result<(String, String, std::process::ExitStatus), String> {
95        let mut cmd = Command::new("agy");
96        cmd.args(&cmd_args[1..])
97            .current_dir(workspace)
98            .envs(env)
99            .stdin(Stdio::null())
100            .stdout(Stdio::piped())
101            .stderr(Stdio::piped())
102            .kill_on_drop(true);
103        #[cfg(unix)]
104        cmd.process_group(0);
105
106        let child = cmd.spawn()
107            .map_err(|e| format!("Failed to spawn agy command: {}", e))?;
108
109        let res = match tokio::time::timeout(tokio::time::Duration::from_secs(300), child.wait_with_output()).await {
110            Ok(Ok(o)) => Ok((
111                String::from_utf8_lossy(&o.stdout).into(),
112                String::from_utf8_lossy(&o.stderr).into(),
113                o.status,
114            )),
115            Ok(Err(e)) => Err(format!("Failed to wait for agy: {}", e)),
116            Err(_) => Err("Command timed out after 300 seconds".to_string()),
117        };
118
119        res
120    }

122    fn resolve_result_text(&self, ws: &Path, env: &HashMap<String, String>, stdout: &str, brain: Option<&Path>) -> String {
123        events::read_transcript_answer(ws, Some(env), brain)
124            .unwrap_or_else(|| events::parse_antigravity_json(stdout))
125    }

130#[async_trait]
131impl AgentProvider for AntigravityCli {
132    async fn send(
133        &self,
134        prompt: &str,
135        resume_session: Option<&str>,
136        continue_session: bool,
137        _workspace: PathBuf,
138    ) -> Result<CliResponse, String> {
139        let env = self.build_env();
140        let agy_ws = self.agy_workspace();
141
142        crate::cli::antigravity::trust::trust_workspace_in_settings(&agy_ws, None);
143
144        let (stdout_str, stderr_str, status) = if let Some(session_id) = resume_session {
145            self.run_in_pty_session(session_id, prompt, &env, &agy_ws).await?
146        } else {
147            let cmd_args = self.build_command(prompt, resume_session, continue_session);
148            self.run_oneshot(&cmd_args, &env, &agy_ws).await?
149        };
150
151        let final_session_id = if let Some(sid) = resume_session {
152            Some(sid.to_string())
153        } else {
154            events::resolve_brain_dir(&agy_ws, Some(&env))
155                .as_ref()
156                .and_then(|d| d.file_name())
157                .map(|name| name.to_string_lossy().to_string())
158        };
159
160        let resolved_brain_dir = if let Some(ref sid) = final_session_id {
161            Some(events::agy_state_root(Some(&env)).join("brain").join(sid))
162        } else {
163            events::resolve_brain_dir(&agy_ws, Some(&env))
164        };
165
166        let result_text = self.resolve_result_text(&agy_ws, &env, &stdout_str, resolved_brain_dir.as_deref());
167
168        if resume_session.is_none() {
169            if let Some(ref fid) = final_session_id {
170                self.ensure_interactive_session(fid, &agy_ws, &env).await?;
171            }
172        }
173
174        let is_error = !status.success();
175        Ok(CliResponse {
176            session_id: final_session_id,
177            result: result_text,
178            is_error,
179            returncode: status.code(),
180            stderr: stderr_str,
181        })
182    }

184    async fn send_streaming<'a>(
185        &'a self,
186        prompt: &str,
187        resume_session: Option<&str>,
188        continue_session: bool,
189        workspace: PathBuf,
190    ) -> Result<BoxStream<'a, StreamEvent>, String> {
191        let (tx, rx) = tokio::sync::mpsc::unbounded_channel();
192        let self_arc = std::sync::Arc::new(self.clone());
193        let prompt_string = prompt.to_string();
194        let resume_id = resume_session.map(|s| s.to_string());
195        let workspace_path = workspace.clone();
196
197        let env = self.build_env();
198        let initial_size = resume_id.as_ref().map(|sid| {
199            let p = events::agy_state_root(Some(&env)).join("brain").join(sid)
200                .join(".system_generated/logs/transcript_full.jsonl");
201            std::fs::metadata(p).map(|m| m.len()).unwrap_or(0)
202        }).unwrap_or(0);
203
204        let rid = resume_id.clone();
205        let oneshot_handle = tokio::spawn(async move {
206            self_arc.send(&prompt_string, rid.as_deref(), continue_session, workspace_path).await
207        });
208
209        super::polling::spawn_log_polling(oneshot_handle, tx, self.agy_workspace(), self.build_env(), Some(initial_size), resume_id);
210
211        let stream = futures::stream::unfold(rx, |mut rx| async move {
212            rx.recv().await.map(|event| (event, rx))
213        });
214        Ok(Box::pin(stream))
215    }
216}

218async fn wait_for_pty_prompt(mgr: &super::session::SessionManager, sid: &str) -> Result<(), String> {
219    let start = std::time::Instant::now();
220    while start.elapsed().as_secs() < 15 {
221        let mut hs = mgr.holders.lock().await;
222        let h = hs.get_mut(sid).ok_or("No holder")?;
223        let out = h.output.lock().await.clone();
224        let dead = h.child.try_wait().ok().flatten().is_some();
225        let s = String::from_utf8_lossy(&out);
226        if s.contains("\r> ") || s.contains("\n> ") || s.contains("\u{1b}[?25h") || dead {
227            return Ok(());
228        }
229        drop(hs);
230        tokio::time::sleep(tokio::time::Duration::from_millis(100)).await;
231    }
232    Err("Timeout waiting for interactive session initialization".to_string())
233}
