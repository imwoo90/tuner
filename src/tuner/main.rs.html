1//! # Tuner Daemon Application Entry Point (main.rs)
2//!
3//! ## Overview
4//! The primary boot orchestrator for the `tuner` service. Operates as a supervisor master
5//! spawning individual profile workers or as a worker managing bots.
6//!
7//! ## Application Architecture
8//! - **Master Mode**: Supervises child profile workers, restarting them on failure, and checks restart markers.
9//! - **Worker Mode**: Runs the bot event loop, schedules cron tasks, and starts webhook ports.
10//!
11//! ## Search Tags
12//! #main-entry, #master-mode, #worker-mode, #process-supervisor

93fn prompt_input(query: &str) -> Result<String, String> {
94    print!("{}", query);
95    std::io::stdout().flush().map_err(|e| e.to_string())?;
96    let mut input = String::new();
97    std::io::stdin().read_line(&mut input).map_err(|e| e.to_string())?;
98    Ok(input.trim().to_string())
99}

101fn run_setup_wizard(paths: &workspace::paths::DuctorPaths) -> Result<(), String> {
102    println!("🤖 [tuner] Setup Wizard");
103    workspace::init::init_workspace(paths)?;
104    
105    let token = loop {
106        let t = prompt_input("Enter your Telegram Bot Token: ")?;
107        if !t.is_empty() && t != "YOUR_BOT_TOKEN_HERE" {
108            break t;
109        }
110        println!("❌ Token cannot be empty. Please try again.");
111    };
112
113    let content = std::fs::read_to_string(&paths.config_path()).map_err(|e| e.to_string())?;
114    let mut config_val: serde_json::Value = serde_json::from_str(&content).map_err(|e| e.to_string())?;
115    
116    if let Some(obj) = config_val.as_object_mut() {
117        obj.insert("telegram_token".to_string(), serde_json::Value::String(token));
118        obj.insert("allowed_user_ids".to_string(), serde_json::Value::Array(Vec::new()));
119    }
120
121    let config_content = serde_json::to_string_pretty(&config_val).map_err(|e| e.to_string())?;
122    std::fs::write(&paths.config_path(), config_content).map_err(|e| e.to_string())?;
123    println!("✅ Configuration saved successfully to {:?}", paths.config_path());
124
125    let config = config::CliConfig::load_from_file(&paths.config_path())?;
126
127    let install_sys = prompt_input("Do you want to install tuner as a systemd user service? (y/n): ")?;
128    if install_sys.to_lowercase().starts_with('y') {
129        setup::install_systemd_service(&config)?;
130    }
131    
132    println!("🎉 Setup complete! Please start the tuner bot and send a message (e.g. /start) to register as owner.");
133    Ok(())
134}

138async fn spawn_worker(current_exe: &std::path::Path, name: &str) -> Result<tokio::process::Child, String> {
139    tokio::process::Command::new(current_exe)
140        .arg("--worker")
141        .arg(name)
142        .env_remove("TELEGRAM_TOKEN")
143        .kill_on_drop(true)
144        .spawn()
145        .map_err(|e| format!("Failed to spawn worker '{}': {}", name, e))
146}

148async fn check_restart_marker(marker: &std::path::Path, workers: &mut [(String, tokio::process::Child)]) {
149    if marker.exists() {
150        let _ = std::fs::remove_file(marker);
151        println!("🤖 [tuner] Master detected restart request. Terminating all workers...");
152        for (name, child) in workers.iter_mut() {
153            println!("🤖 [tuner] Killing worker: {}", name);
154            let _ = child.kill().await;
155        }
156        std::process::exit(42);
157    }
158}

160async fn monitor_workers(workers: &mut [(String, tokio::process::Child)], current_exe: &std::path::Path) -> Result<(), String> {
161    let mut exit_requested = None;
162    for (name, child) in workers.iter_mut() {
163        match child.try_wait() {
164            Ok(Some(status)) => {
165                println!("🤖 [tuner] Worker '{}' exited with status: {}", name, status);
166                if let Some(code) = status.code() {
167                    if code == 42 {
168                        exit_requested = Some(name.clone());
169                        break;
170                    }
171                }
172                println!("🤖 [tuner] Restarting worker: {}", name);
173                *child = spawn_worker(current_exe, name).await?;
174            }
175            Ok(None) => {}
176            Err(e) => {
177                eprintln!("🤖 [tuner] Error checking status of worker '{}': {}", name, e);
178            }
179        }
180    }
181
182    if let Some(ref requester) = exit_requested {
183        println!("🤖 [tuner] Worker '{}' requested restart. Exiting master...", requester);
184        for (name, child) in workers.iter_mut() {
185            if name != requester {
186                let _ = child.kill().await;
187            }
188        }
189        std::process::exit(42);
190    }
191    Ok(())
192}

194async fn run_master_mode(config: config::CliConfig) -> Result<(), String> {
195    println!("🤖 [tuner] Starting in Master Mode (Supervisor)...");
196    let current_exe = std::env::current_exe()
197        .map_err(|e| format!("Failed to resolve current binary path: {}", e))?;
198
199    let mut workers = Vec::new();
200    for profile in &config.profiles {
201        println!("🤖 [tuner] Spawning worker for profile: {}", profile.name);
202        let child = spawn_worker(&current_exe, &profile.name).await?;
203        workers.push((profile.name.clone(), child));
204    }
205
206    let home = std::env::var("HOME").unwrap_or_else(|_| ".".to_string());
207    let marker = std::path::PathBuf::from(home).join(".tuner/restart-requested");
208    let mut interval = tokio::time::interval(tokio::time::Duration::from_secs(1));
209
210    loop {
211        interval.tick().await;
212        check_restart_marker(&marker, &mut workers).await;
213        monitor_workers(&mut workers, &current_exe).await?;
214    }
215}

219async fn handle_early_args(args: &[String]) -> Result<Option<()>, String> {
220    if args.contains(&"--version".to_string()) || args.contains(&"-V".to_string()) {
221        println!("tuner {}", env!("CARGO_PKG_VERSION"));
222        return Ok(Some(()));
223    }
224    if args.contains(&"--upgrade".to_string()) {
225        upgrade::run_cli_upgrade().await?;
226        return Ok(Some(()));
227    }
228    Ok(None)
229}

231async fn handle_daemon_mode(args: &[String], config: &config::CliConfig) -> Result<Option<()>, String> {
232    if args.contains(&"--install-systemd".to_string()) {
233        setup::install_systemd_service(config)?;
234        return Ok(Some(()));
235    }
236    if args.contains(&"--supervisor".to_string()) {
237        let current_exe = std::env::current_exe()
238            .map_err(|e| format!("Failed to resolve current binary path: {}", e))?;
239        let filtered_args: Vec<String> = args.iter().cloned().skip(1).filter(|arg| arg != "--supervisor").collect();
240        let supervisor = supervisor::Supervisor::with_args(current_exe, filtered_args);
241        supervisor.run().await?;
242        return Ok(Some(()));
243    }
244    Ok(None)
245}

247#[tokio::main]
248async fn main() -> Result<(), String> {
249    let args: Vec<String> = std::env::args().collect();
250
251    if let Some(()) = handle_early_args(&args).await? {
252        return Ok(());
253    }
254
255    let worker_profile = args.iter().position(|arg| arg == "--worker")
256        .and_then(|pos| args.get(pos + 1).cloned());
257
258    let tuner_home = std::path::PathBuf::from(std::env::var("HOME").unwrap_or_else(|_| ".".to_string())).join(".tuner");
259    let paths = workspace::paths::resolve_paths(Some(tuner_home), None, None, worker_profile.clone());
260
261    if args.contains(&"--setup".to_string()) {
262        return run_setup_wizard(&paths);
263    }
264
265    let _ = setup::load_env_file(&paths.env_file());
266    workspace::init::init_workspace(&paths)?;
267
268    let mut config = config::CliConfig::load_from_file(&paths.config_path())?;
269    config.working_dir = paths.workspace().clone();
270
271    if let Some(ref profile_name) = worker_profile {
272        setup::override_profile_config(&mut config, profile_name, &paths)?;
273        let profile_config_path = paths.profile_home().join("config").join("config.json");
274        if profile_config_path.is_file() {
275            config.merge_profile_file(&profile_config_path)?;
276        }
277    }
278
279    let startup_lang = config.language.as_deref().unwrap_or("en");
280    i18n::init(startup_lang);
281
282    if let Some(()) = handle_daemon_mode(&args, &config).await? {
283        return Ok(());
284    }
285
286    if let Some(ref p) = worker_profile {
287        println!("🤖 [tuner] Starting Telegram bot worker for '{}'...", p);
288        telegram::run_bot(config, paths).await
289    } else {
290        run_master_mode(config).await
291    }
292}
