57pub fn from_background_result(result: &BackgroundResult) -> Envelope {
58    let status_str = match result.status {
59        BackgroundResultStatus::Success => "success".to_string(),
60        BackgroundResultStatus::Aborted => "aborted".to_string(),
61        BackgroundResultStatus::ErrorCli => "error:cli".to_string(),
62        BackgroundResultStatus::ErrorTimeout => "error:timeout".to_string(),
63        BackgroundResultStatus::ErrorCliNotFound => "error:cli_not_found".to_string(),
64        BackgroundResultStatus::ErrorInternal => "error:internal".to_string(),
65    };
66    let is_error = status_str.starts_with("error:");
67    let mut env = Envelope::new(Origin::Background, result.chat_id);
68    env.reply_to_message_id = Some(result.message_id);
69    env.thread_id = result.thread_id;
70    env.topic_id = result.thread_id;
71    env.prompt_preview = result.prompt_preview.clone();
72    env.result_text = result.result_text.clone();
73    env.status = status_str;
74    env.is_error = is_error;
75    env.elapsed_seconds = result.elapsed_seconds;
76    env
77}

79pub fn from_cron_result(
80    title: &str,
81    result_text: &str,
82    status: &str,
83    chat_id: Option<i64>,
84    topic_id: Option<i64>,
85    transport: Option<&str>,
86) -> Envelope {
87    let mut env = Envelope::new(Origin::Cron, chat_id.unwrap_or(0));
88    env.topic_id = topic_id;
89    env.result_text = result_text.to_string();
90    env.status = status.to_string();
91    env.is_error = status.to_lowercase().contains("error");
92    if chat_id.is_some() {
93        env.delivery = DeliveryMode::Unicast;
94    } else {
95        env.delivery = DeliveryMode::Broadcast;
96    }
97    if let Some(t) = transport {
98        env.transport = t.to_string();
99    }
100    env.metadata.insert("title".to_string(), title.to_string());
101    env
102}

139pub fn from_interagent_result(
140    result: &InterAgentResult,
141    chat_id: i64,
142    injection_prompt: Option<&str>,
143    transport: Option<&str>,
144) -> Envelope {
145    let target_chat_id = if result.chat_id != 0 { result.chat_id } else { chat_id };
146    let mut env = Envelope::new(Origin::Interagent, target_chat_id);
147    env.topic_id = result.topic_id;
148    env.result_text = result.result_text.clone();
149    env.status = if result.success { "success".to_string() } else { "error".to_string() };
150    env.is_error = !result.success;
151    env.lock_mode = if result.success { LockMode::Required } else { LockMode::None };
152    
153    if result.success && injection_prompt.is_some() {
154        env.needs_injection = true;
155        env.prompt = injection_prompt.unwrap().to_string();
156    }
157
158    env.metadata.insert("sender".to_string(), result.sender.clone());
159    env.metadata.insert("recipient".to_string(), result.recipient.clone());
160    env.metadata.insert("task_id".to_string(), result.task_id.clone());
161    if let Some(ref err) = result.error {
162        env.metadata.insert("error".to_string(), err.clone());
163    }
164    if let Some(t) = transport {
165        env.transport = t.to_string();
166        env.metadata.insert("transport".to_string(), t.to_string());
167    }
168    env
169}

171pub fn from_task_result(result: &TaskResult) -> Envelope {
172    let mut env = Envelope::new(Origin::TaskResult, result.chat_id);
173    env.topic_id = result.thread_id;
174    env.status = result.status.clone();
175    env.is_error = result.status == "failed";
176    env.lock_mode = if result.status == "cancelled" { LockMode::None } else { LockMode::Required };
177    env.needs_injection = result.status != "cancelled";
178    
179    if env.needs_injection {
180        env.prompt = if result.status == "done" {
181            format!(
182                "[BACKGROUND TASK COMPLETED]\ntask_id='{}'\nresult:\n{}\nReview this result critically.",
183                result.task_id, result.result_text
184            )
185        } else {
186            format!(
187                "[BACKGROUND TASK FAILED]\ntask_id='{}'\nerror:\n{}\nReview this failure.",
188                result.task_id, result.error
189            )
190        };
191    }
192
193    env.elapsed_seconds = result.elapsed_seconds;
194    env.provider = result.provider.clone();
195    env.model = result.model.clone();
196    env.session_id = result.session_id.clone();
197
198    env.metadata.insert("name".to_string(), result.name.clone());
199    env.metadata.insert("parent_agent".to_string(), result.parent_agent.clone());
200    if !result.error.is_empty() {
201        env.metadata.insert("error".to_string(), result.error.clone());
202    }
203    env
204}

206pub fn from_task_question(
207    task_id: &str,
208    question: &str,
209    preview: &str,
210    chat_id: i64,
211    topic_id: Option<i64>,
212) -> Envelope {
213    let mut env = Envelope::new(Origin::TaskQuestion, chat_id);
214    env.topic_id = topic_id;
215    env.prompt = question.to_string();
216    env.prompt_preview = preview.to_string();
217    env.lock_mode = LockMode::Required;
218    env.needs_injection = true;
219    env.metadata.insert("task_id".to_string(), task_id.to_string());
220    env
221}

223pub fn from_user_message(
224    chat_id: i64,
225    text: &str,
226    topic_id: Option<i64>,
227    source: Option<Origin>,
228) -> Envelope {
229    let mut env = Envelope::new(source.unwrap_or(Origin::User), chat_id);

232    env.prompt_preview = text.chars().take(80).collect();
