15pub fn load_env_file(path: &Path) -> Result<(), String> {
16    if !path.is_file() {
17        return Ok(());
18    }
19    let content = std::fs::read_to_string(path).map_err(|e| e.to_string())?;
20    for line in content.lines() {
21        let trimmed = line.trim();
22        if trimmed.is_empty() || trimmed.starts_with('#') {
23            continue;
24        }
25        let line_to_parse = if trimmed.starts_with("export ") {
26            trimmed["export ".len()..].trim()
27        } else {
28            trimmed
29        };
30        if let Some(pos) = line_to_parse.find('=') {
31            let key = line_to_parse[..pos].trim();
32            let mut val = line_to_parse[pos + 1..].trim().to_string();
33            if (val.starts_with('"') && val.ends_with('"')) || (val.starts_with('\'') && val.ends_with('\'')) {
34                if val.len() >= 2 {
35                    val = val[1..val.len() - 1].to_string();
36                }
37            }
38            if !key.is_empty() && std::env::var(key).is_err() {
39                unsafe { std::env::set_var(key, val); }
40            }
41        }
42    }
43    Ok(())
44}

46pub fn install_systemd_service(config: &CliConfig) -> Result<(), String> {
47    let home = std::env::var("HOME").unwrap_or_else(|_| ".".to_string());
48    let current_exe = std::env::current_exe().map_err(|e| e.to_string())?;
49    let project_root = current_exe.parent().and_then(|p| p.parent()).and_then(|p| p.parent())
50        .map(|p| p.to_path_buf())
51        .unwrap_or_else(|| std::env::current_dir().unwrap_or_else(|_| std::path::PathBuf::from(".")));
52    let path_env = std::env::var("PATH").unwrap_or_else(|_| "/usr/local/bin:/usr/bin:/bin".to_string());
53
54    let token_line = if let Ok(tok) = std::env::var("TELEGRAM_TOKEN") {
55        format!("Environment=\"TELEGRAM_TOKEN={}\"\n", tok)
56    } else if !config.telegram_token.is_empty() && config.telegram_token != "YOUR_BOT_TOKEN_HERE" {
57        format!("Environment=\"TELEGRAM_TOKEN={}\"\n", config.telegram_token)
58    } else {
59        String::new()
60    };
61
62    let unit_content = format!(
63        "[Unit]\nDescription=Tuner Bot\nAfter=network.target\n\n[Service]\nType=simple\n\
64         WorkingDirectory={}\nExecStart={}\n{}Environment=\"HOME={}\"\nEnvironment=\"PATH={}\"\n\
65         Restart=always\nRestartSec=10\n\n\
66         [Install]\nWantedBy=default.target\n",
67        project_root.to_string_lossy(), current_exe.to_string_lossy(), token_line, home, path_env
68    );
69    let systemd_dir = std::path::PathBuf::from(&home).join(".config/systemd/user");
70    std::fs::create_dir_all(&systemd_dir).map_err(|e| e.to_string())?;
71    let service_file = systemd_dir.join("tuner.service");
72    std::fs::write(&service_file, unit_content).map_err(|e| e.to_string())?;
73    println!("🤖 [tuner] Installed successfully to {:?}", service_file);
74    println!("💡 Run: systemctl --user daemon-reload && systemctl --user restart tuner");
75    Ok(())
76}

78pub fn override_profile_config(
79    config: &mut CliConfig,
80    profile_name: &str,
81    paths: &crate::workspace::paths::DuctorPaths,
82) -> Result<(), String> {
83    let profile_cfg = config.profiles.iter().find(|p| p.name == profile_name)
84        .ok_or_else(|| format!("Profile '{}' not found in config.json", profile_name))?;
85    if !profile_cfg.telegram_token.is_empty() && profile_cfg.telegram_token != "YOUR_BOT_TOKEN_HERE" && !profile_cfg.telegram_token.starts_with("YOUR_") {
86        config.telegram_token = profile_cfg.telegram_token.clone();
87    } else if profile_name != "default" {
88        config.telegram_token = String::new();
89    }
90    if !profile_cfg.allowed_user_ids.is_empty() && profile_cfg.allowed_user_ids != vec![123456789] {
91        config.allowed_user_ids = profile_cfg.allowed_user_ids.clone();
92    }
93    if !profile_cfg.allowed_group_ids.is_empty() && profile_cfg.allowed_group_ids != vec![-1001234567890] {
94        config.allowed_group_ids = profile_cfg.allowed_group_ids.clone();
95    }
96    config.working_dir = profile_cfg.working_dir.clone().unwrap_or_else(|| paths.workspace());
97    if let Some(ref m) = profile_cfg.model {
98        config.model = Some(m.clone());
99    }
100    if let Some(ref p) = profile_cfg.system_prompt {
101        config.system_prompt = Some(p.clone());
102    }
103    if let Some(ref p) = profile_cfg.append_system_prompt {
104        config.append_system_prompt = Some(p.clone());
105    }
106    if let Some(ref l) = profile_cfg.language {
107        config.language = Some(l.clone());
108    }
109    Ok(())
110}
