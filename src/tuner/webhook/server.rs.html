24pub type DispatchHandler = Arc<dyn Fn(String, serde_json::Value) + Send + Sync + 'static>;

26pub struct ServerState {
27    pub manager: Arc<WebhookManager>,
28    pub rate_limiter: RateLimiter,
29    pub global_token: String,
30    pub dispatch: Option<DispatchHandler>,
31    pub max_body_bytes: usize,
32}
33
34pub struct WebhookServer {
35    state: Arc<ServerState>,
36    shutdown_tx: std::sync::Mutex<Option<tokio::sync::oneshot::Sender<()>>>,
37}
38
39impl WebhookServer {
40    pub fn new(
41        manager: Arc<WebhookManager>,
42        rate_limit_per_minute: usize,
43        global_token: String,
44        dispatch: Option<DispatchHandler>,
45        max_body_bytes: usize,
46    ) -> Self {
47        Self {
48            state: Arc::new(ServerState {
49                manager,
50                rate_limiter: RateLimiter::new(rate_limit_per_minute),
51                global_token,
52                dispatch,
53                max_body_bytes,
54            }),
55            shutdown_tx: std::sync::Mutex::new(None),
56        }
57    }
58
59    pub async fn start(&self, host: &str, port: u16) -> Result<(), String> {
60        let (tx, rx) = tokio::sync::oneshot::channel::<()>();
61        *self.shutdown_tx.lock().unwrap() = Some(tx);
62
63        let app = Router::new()
64            .route("/health", get(handle_health))
65            .route("/hooks/:hook_id", post(handle_webhook))
66            .layer(DefaultBodyLimit::max(self.state.max_body_bytes))
67            .with_state(self.state.clone());
68
69        let addr = format!("{}:{}", host, port);
70        let listener = tokio::net::TcpListener::bind(&addr)
71            .await
72            .map_err(|e| e.to_string())?;
73
74        let server_future = axum::serve(listener, app).with_graceful_shutdown(async move {
75            let _ = rx.await;
76        });
77
78        tokio::spawn(async move {
79            if let Err(e) = server_future.await {
80                eprintln!("Webhook Axum server error: {}", e);
81            }
82        });
83
84        Ok(())
85    }
86
87    pub fn stop(&self) {
88        if let Some(tx) = self.shutdown_tx.lock().unwrap().take() {
89            let _ = tx.send(());
90        }
91    }
92}

94async fn handle_health() -> impl IntoResponse {
95    (StatusCode::OK, Json(serde_json::json!({ "status": "ok" })))
96}

98async fn handle_webhook(
99    Path(hook_id): Path<String>,
100    State(state): State<Arc<ServerState>>,
101    headers: HeaderMap,
102    body: axum::body::Bytes,
103) -> impl IntoResponse {
104    if !state.rate_limiter.check().await {
105        return (StatusCode::TOO_MANY_REQUESTS, Json(serde_json::json!({ "error": "rate_limited" }))).into_response();
106    }
107    let ct = headers.get(header::CONTENT_TYPE).and_then(|v| v.to_str().ok()).unwrap_or("");
108    if ct != "application/json" {
109        return (StatusCode::UNSUPPORTED_MEDIA_TYPE, Json(serde_json::json!({ "error": "content_type_must_be_json" }))).into_response();
110    }
111    let payload: serde_json::Value = match serde_json::from_slice::<serde_json::Value>(&body) {
112        Ok(v) if v.is_object() => v,
113        Ok(_) => return (StatusCode::BAD_REQUEST, Json(serde_json::json!({ "error": "body_must_be_object" }))).into_response(),
114        _ => return (StatusCode::BAD_REQUEST, Json(serde_json::json!({ "error": "invalid_json" }))).into_response(),
115    };
116    let Some(hook) = state.manager.get_hook(&hook_id).await else {
117        return (StatusCode::NOT_FOUND, Json(serde_json::json!({ "error": "hook_not_found" }))).into_response();
118    };
119    if !hook.enabled {
120        return (StatusCode::FORBIDDEN, Json(serde_json::json!({ "error": "hook_disabled" }))).into_response();
121    }
122    if !check_auth(&headers, &hook, &body, &state.global_token) {
123        return (StatusCode::UNAUTHORIZED, Json(serde_json::json!({ "error": "unauthorized" }))).into_response();
124    }
125    if let Some(ref dispatch) = state.dispatch {
126        let dispatch_clone = dispatch.clone();
127        let hook_id_clone = hook_id.clone();
128        let payload_clone = payload.clone();
129        tokio::spawn(async move {
130            dispatch_clone(hook_id_clone, payload_clone);
131        });
132    }
133    (StatusCode::ACCEPTED, Json(serde_json::json!({ "accepted": true, "hook_id": hook_id }))).into_response()
134}

136fn check_auth(headers: &HeaderMap, hook: &crate::webhook::models::WebhookEntry, body: &[u8], global_token: &str) -> bool {
137    let auth_header = headers.get(header::AUTHORIZATION).and_then(|v| v.to_str().ok()).unwrap_or("");
138    let sig_val = if !hook.hmac_header.is_empty() {
139        headers.get(&hook.hmac_header).and_then(|v| v.to_str().ok()).unwrap_or("")
140    } else {
141        ""
142    };
143    crate::webhook::auth::validate_hook_auth(hook, auth_header, sig_val, body, global_token)
144}
