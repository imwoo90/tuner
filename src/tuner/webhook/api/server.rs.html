42impl ApiServer {
43    pub fn new(config: crate::config::ApiConfig, default_chat_id: i64) -> Self {
44        Self {
45            state: Arc::new(std::sync::Mutex::new(ApiServerState {
46                config,
47                default_chat_id,
48                lock_pool: Arc::new(crate::bus::lock_pool::LockPool::new_default()),
49                message_handler: None,
50                abort_handler: None,
51                allowed_roots: None,
52                upload_dir: None,
53                workspace: None,
54                provider_info: serde_json::json!([]),
55                active_state_getter: None,
56                next_conn_id: 0,
57                active_ws: Arc::new(std::sync::Mutex::new(std::collections::HashMap::new())),
58
59            })),
60            shutdown_tx: std::sync::Mutex::new(None),
61        }
62    }

102    pub async fn start(&self, host: &str, port: u16) -> Result<(), String> {
103        let (tx, rx) = tokio::sync::oneshot::channel::<()>();
104        *self.shutdown_tx.lock().unwrap() = Some(tx);
105
106        let app = Router::new()
107            .route("/health", get(handle_health))
108            .route("/ws", get(handle_websocket))
109            .route("/files", get(handle_file_download))
110            .route("/upload", post(handle_file_upload))
111
112            .layer(DefaultBodyLimit::max(50 * 1024 * 1024))
113            .with_state(self.state.clone());
114
115        let addr = format!("{}:{}", host, port);
116        let listener = tokio::net::TcpListener::bind(&addr)
117            .await
118            .map_err(|e| e.to_string())?;
119
120        let server_future = axum::serve(listener, app).with_graceful_shutdown(async move {
121            let _ = rx.await;
122        });
123
124        tokio::spawn(async move {
125            if let Err(e) = server_future.await {
126                eprintln!("API Axum server error: {}", e);
127            }
128        });
129
130        Ok(())
131    }
132
133    pub async fn stop(&self) {
134        if let Some(tx) = self.shutdown_tx.lock().unwrap().take() {
135            let _ = tx.send(());
136        }
137        let active = {
138            let s = self.state.lock().unwrap();
139            s.active_ws.lock().unwrap().clone()
140        };
141        for (_, ws_tx) in active {
142            let _ = ws_tx.send(axum::extract::ws::Message::Close(None));
143        }
144    }
145}
