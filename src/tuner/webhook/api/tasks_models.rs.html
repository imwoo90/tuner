1//! # Webhook Task Serialization Models

9use serde::{Deserialize, Serialize};
10
11#[derive(Debug, Deserialize)]
12pub struct TaskCreateRequest {
13    #[serde(rename = "from")]
14    pub parent_agent: String,
15    pub prompt: String,
16    pub name: Option<String>,
17    pub chat_id: Option<i64>,
18    pub thread_id: Option<i64>,
19    pub provider: Option<String>,
20    pub model: Option<String>,
21    pub thinking: Option<String>,
22    pub priority: Option<String>,
23    #[serde(default)]
24    pub depends_on: Vec<String>,
25}
26
27#[derive(Debug, Serialize)]
28pub struct TaskCreateResponse {
29    pub success: bool,
30    #[serde(skip_serializing_if = "Option::is_none")]
31    pub task_id: Option<String>,
32    #[serde(skip_serializing_if = "Option::is_none")]
33    pub error: Option<String>,
34}
35
36#[derive(Debug, Deserialize)]
37pub struct TaskResumeRequest {
38    pub task_id: String,
39    pub prompt: String,
40    #[serde(rename = "from")]
41    pub parent_agent: String,
42}
43
44#[derive(Debug, Deserialize)]
45pub struct TaskAskParentRequest {

47    pub question: String,
48}
49
50#[derive(Debug, Serialize)]
51pub struct TaskAskParentResponse {
52    pub success: bool,
53    pub answer: String,
54    #[serde(skip_serializing_if = "Option::is_none")]
55    pub error: Option<String>,
56}
57
58#[derive(Debug, Deserialize)]
59pub struct TaskCancelRequest {
60    pub task_id: String,
61    #[serde(rename = "from")]
62    pub parent_agent: String,
63}
64
65#[derive(Debug, Deserialize)]
66pub struct TaskDeleteRequest {
67    pub task_id: String,
68    #[serde(rename = "from")]
69    pub parent_agent: String,
70}
71
72#[derive(Debug, Serialize)]
73pub struct StandardResponse {
74    pub success: bool,

77}
78

83
84impl TaskCreateRequest {
85    pub fn into_submit(self) -> crate::tasks::TaskSubmit {
86        crate::tasks::TaskSubmit {
87            chat_id: self.chat_id.unwrap_or(0),
88            prompt: self.prompt,
89            message_id: 0,
90            thread_id: self.thread_id,
91            parent_agent: self.parent_agent,
92            name: self.name.unwrap_or_default(),
93            provider_override: self.provider.unwrap_or_default(),
94            model_override: self.model.unwrap_or_default(),
95            thinking_override: self.thinking.unwrap_or_default(),
96            priority: self.priority.unwrap_or_default(),
97            depends_on: self.depends_on,
98        }
99    }
