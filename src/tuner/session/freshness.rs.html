1//! # Session Freshness Checker
2//!
3//! ## Overview
4//! Implements logical rules to verify if an active session is still fresh, checking inactivity timeouts,
5//! max message limits, and daily user-timezone resets.
6//!
7//! ## Collaboration Graph
8//! - Called by [`SessionManager::is_fresh`](super::manager::SessionManager::is_fresh) on new message ingress.
9//!
10//! ## Search Tags
11//! #session-freshness, #timeout-inactivity, #daily-resets, #timezone-calculator
12
13use chrono::{DateTime, Utc, Duration, LocalResult, TimeZone};
14use chrono_tz::Tz;
15use crate::session::data::SessionData;
16
17/// Evaluate whether the given session has crossed the timezone-based daily reset threshold.
18pub fn has_crossed_daily_reset(
19    last_active: &DateTime<Utc>,
20    now: &DateTime<Utc>,
21    user_timezone: &str,
22    daily_reset_hour: u32,
23) -> bool {
24    let tz = user_timezone.parse::<Tz>().unwrap_or(Tz::UTC);
25    let now_local = now.with_timezone(&tz);
26    let last_local = last_active.with_timezone(&tz);
27
28    let today_reset = match tz.from_local_datetime(
29        &now_local.date_naive().and_hms_opt(daily_reset_hour, 0, 0).unwrap()
30    ) {
31        LocalResult::Single(dt) => dt,
32        LocalResult::Ambiguous(dt1, _) => dt1,
33        LocalResult::None => return true, // Expire on error
34    };
35
36    if now_local >= today_reset {
37        last_local < today_reset
38    } else {
39        let yesterday_reset = today_reset - Duration::days(1);
40        last_local < yesterday_reset
41    }
42}

44/// Check if the session is fresh according to max messages, idle timeout, and daily reset rules.
45pub fn is_session_fresh(
46    session: &SessionData,
47    max_session_messages: Option<i64>,
48    idle_timeout_minutes: i64,
49    daily_reset_enabled: bool,
50    daily_reset_hour: u32,
51    user_timezone: &str,
52) -> bool {
53    let now = Utc::now();
54    let last = match DateTime::parse_from_rfc3339(&session.last_active) {
55        Ok(dt) => dt.with_timezone(&Utc),
56        Err(_) => return false,
57    };
58
59    if let Some(max_msg) = max_session_messages {
60        let current_provider_ps = session.provider_sessions.get(&session.provider);
61        let msg_count = current_provider_ps.map(|ps| ps.message_count).unwrap_or(0);
62        if msg_count >= max_msg {
63            return false;
64        }
65    }
66
67    if idle_timeout_minutes > 0 && now.signed_duration_since(last) >= Duration::minutes(idle_timeout_minutes) {
68        return false;
69    }
70
71    if daily_reset_enabled && has_crossed_daily_reset(&last, &now, user_timezone, daily_reset_hour) {
72        return false;
73    }
74
75    true
76}
