1//! # Session Storage Key Identifier
2//!
3//! ## Overview
4//! Defines the [`SessionKey`] representation. Matches Telegram chat/topic ID mappings or webhook channels
5//! to guarantee strict separation of concurrent workspaces.
6//!
7//! ## Collaboration Graph
8//! - Constructed inside message routers (e.g. [`process_text_with_files`](crate::telegram::process_text_with_files)).
9//! - Used by [`SessionManager`](super::manager::SessionManager) as hash keys.
10//!
11//! ## Search Tags
12//! #session-keys, #telegram-keys, #webhook-keys, #key-serialization
13
14use std::str::FromStr;
15
16/// Composite session identifier: transport + chat + optional topic/channel.
17#[derive(Clone, Debug, Hash, PartialEq, Eq, serde::Serialize, serde::Deserialize)]
18pub struct SessionKey {
19    pub transport: String,
20    pub chat_id: i64,
21    pub topic_id: Option<i64>,
22}
23
24impl Default for SessionKey {
25    fn default() -> Self {
26        Self {
27            transport: "tg".to_string(),
28            chat_id: 0,
29            topic_id: None,
30        }
31    }
32}
33
34impl SessionKey {
35    /// Create a new session key for Telegram.
36    pub fn telegram(chat_id: i64, topic_id: Option<i64>) -> Self {
37        Self {
38            transport: "tg".to_string(),
39            chat_id,
40            topic_id,
41        }
42    }
43
44    /// Generate a JSON-serializable key for `sessions.json` persistence.
45    pub fn storage_key(&self) -> String {
46        match self.topic_id {
47            None => format!("{}:{}", self.transport, self.chat_id),
48            Some(tid) => format!("{}:{}:{}", self.transport, self.chat_id, tid),
49        }
50    }
51
52    pub fn lock_key(&self) -> (i64, Option<i64>) {
53        (self.chat_id, self.topic_id)
54    }
55
56    pub fn matrix(chat_id: i64) -> Self {
57        Self {
58            transport: "mx".to_string(),
59            chat_id,
60            topic_id: None,
61        }
62    }
63
64    pub fn for_transport(transport: &str, chat_id: i64, topic_id: Option<i64>) -> Self {
65        Self {
66            transport: transport.to_string(),
67            chat_id,
68            topic_id,
69        }
70    }
71}
72
73impl FromStr for SessionKey {
74    type Err = String;
75
76    fn from_str(s: &str) -> Result<Self, Self::Err> {
77        if s.is_empty() {
78            return Err("Empty key string".to_string());
79        }
80        let p: Vec<&str> = s.split(':').collect();
81        if p.len() > 3 {
82            return Err(format!("Invalid key format: {}", s));
83        }
84        if p.iter().any(|part| part.is_empty()) {
85            return Err(format!("Invalid key format: {}", s));
86        }
87        let err = |name, val: &str| format!("Invalid {}: {}", name, val);
88        match p.len() {
89            1 => Ok(Self {
90                transport: "tg".to_string(),
91                chat_id: p[0].parse().map_err(|_| err("chat_id", p[0]))?,
92                topic_id: None,
93            }),
94            2 => {
95                if let Ok(cid) = p[0].parse::<i64>() {
96                    let tid = p[1].parse().map_err(|_| err("topic_id", p[1]))?;
97                    Ok(Self { transport: "tg".to_string(), chat_id: cid, topic_id: Some(tid) })
98                } else {
99                    let cid = p[1].parse().map_err(|_| err("chat_id", p[1]))?;
100                    Ok(Self { transport: p[0].to_string(), chat_id: cid, topic_id: None })
101                }
102            }
103            3 => {
104                let cid = p[1].parse().map_err(|_| err("chat_id", p[1]))?;
105                let tid = p[2].parse().map_err(|_| err("topic_id", p[2]))?;
106                Ok(Self { transport: p[0].to_string(), chat_id: cid, topic_id: Some(tid) })
107            }
108            _ => Err(format!("Invalid key format: {}", s)),
109        }
110    }
111}

113#[cfg(test)]
114mod tests {
115    use super::*;
116
117    #[test]
118    fn test_storage_key() {
119        let k1 = SessionKey::telegram(12345, None);
120        assert_eq!(k1.storage_key(), "tg:12345");
121
122        let k2 = SessionKey::telegram(12345, Some(99));
123        assert_eq!(k2.storage_key(), "tg:12345:99");
124    }
125
126    #[test]
127    fn test_parse_legacy_keys() {
128        let k1 = SessionKey::from_str("12345").unwrap();
129        assert_eq!(k1.transport, "tg");
130        assert_eq!(k1.chat_id, 12345);
131        assert_eq!(k1.topic_id, None);
132
133        let k2 = SessionKey::from_str("-100123").unwrap();
134        assert_eq!(k2.transport, "tg");
135        assert_eq!(k2.chat_id, -100123);
136        assert_eq!(k2.topic_id, None);
137
138        let k3 = SessionKey::from_str("12345:99").unwrap();
139        assert_eq!(k3.transport, "tg");
140        assert_eq!(k3.chat_id, 12345);
141        assert_eq!(k3.topic_id, Some(99));
142    }
143
144    #[test]
145    fn test_parse_new_prefixed_keys() {
146        let k1 = SessionKey::from_str("tg:12345").unwrap();
147        assert_eq!(k1.transport, "tg");
148        assert_eq!(k1.chat_id, 12345);
149        assert_eq!(k1.topic_id, None);
150
151        let k2 = SessionKey::from_str("tg:12345:99").unwrap();
152        assert_eq!(k2.transport, "tg");
153        assert_eq!(k2.chat_id, 12345);
154        assert_eq!(k2.topic_id, Some(99));
155
156        let k3 = SessionKey::from_str("mx:98765").unwrap();
157        assert_eq!(k3.transport, "mx");
158        assert_eq!(k3.chat_id, 98765);
159        assert_eq!(k3.topic_id, None);
160    }
161
162    #[test]
163    fn test_lock_key() {
164        let k = SessionKey { transport: "mx".to_string(), chat_id: 42, topic_id: Some(7) };
165        assert_eq!(k.lock_key(), (42, Some(7)));
166    }
167
168    #[test]
169    fn test_matrix_factory() {
170        let k = SessionKey::matrix(42);
171        assert_eq!(k.transport, "mx");

173        assert_eq!(k.topic_id, None);

175
176    #[test]
