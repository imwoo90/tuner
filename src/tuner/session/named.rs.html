1//! # Named Session Registry and Alias Resolver
2//!
3//! ## Overview
4//! Resolves human-readable session aliases to raw UUID session strings. Enables resumes
5//! and diagnostics via user-friendly mnemonic names.
6//!
7//! ## Collaboration Graph
8//! - Saves aliases mappings under configuration paths.
9//! - Consulted by Telegram command routing controllers during session switching.
10//!
11//! ## Search Tags
12//! #session-aliases, #name-resolver, #alias-mapping
13
14use std::collections::HashMap;
15use std::fs;
16use std::path::PathBuf;
17use serde::{Deserialize, Serialize};
18use chrono::Utc;
19
20#[derive(Clone, Debug, Serialize, Deserialize, PartialEq)]
21pub struct NamedSession {
22    pub name: String,
23    pub chat_id: i64,
24    pub status: String,
25    pub last_prompt: String,
26}
27
28pub struct NamedSessionRegistry {
29    path: PathBuf,
30    lock: std::sync::Mutex<()>,
31}
32
33fn get_random_seed() -> u64 {
34    let mut seed = Utc::now().timestamp_nanos_opt().unwrap_or(0) as u64;
35    if let Ok(mut f) = std::fs::File::open("/dev/urandom") {
36        let mut buf = [0u8; 8];
37        if std::io::Read::read_exact(&mut f, &mut buf).is_ok() {
38            seed = u64::from_ne_bytes(buf);
39        }
40    }
41    seed
42}
43
44impl NamedSessionRegistry {
45    pub fn new(path: PathBuf) -> Self {
46        Self { path, lock: std::sync::Mutex::new(()) }
47    }
48
49    pub fn load(&self) -> Result<HashMap<String, NamedSession>, String> {
50        if !self.path.exists() { return Ok(HashMap::new()); }
51        let content = fs::read_to_string(&self.path).map_err(|e| e.to_string())?;
52        if content.trim().is_empty() { return Ok(HashMap::new()); }
53        serde_json::from_str(&content).map_err(|e| e.to_string())
54    }
55
56    pub fn save(&self, map: &HashMap<String, NamedSession>) -> Result<(), String> {
57        let temp = self.path.with_extension("tmp");
58        let content = serde_json::to_string_pretty(map).map_err(|e| e.to_string())?;
59        if let Some(parent) = self.path.parent() {
60            fs::create_dir_all(parent).map_err(|e| e.to_string())?;
61        }
62        fs::write(&temp, content).map_err(|e| e.to_string())?;
63        fs::rename(&temp, &self.path).map_err(|e| e.to_string())?;
64        Ok(())
65    }
66
67    fn generate_unique_name(&self, map: &HashMap<String, NamedSession>) -> Result<String, String> {
68        let adjectives = ["bold", "swift", "clever", "quiet", "bright", "wild", "calm", "sharp", "gentle", "proud"];
69        let nouns = ["owl", "fox", "wolf", "bear", "hawk", "deer", "lynx", "hare", "swan", "otter"];
70        let mut seed = get_random_seed();
71        let a: u64 = 6364136223846793005;
72        let c: u64 = 1442695040888963407;
73        for _ in 0..100 {
74            seed = seed.wrapping_mul(a).wrapping_add(c);
75            let adj = adjectives[(seed as usize) % adjectives.len()];
76            seed = seed.wrapping_mul(a).wrapping_add(c);
77            let noun = nouns[(seed as usize) % nouns.len()];
78            let name = format!("{}{}", adj, noun);
79            if !map.contains_key(&name) {
80                return Ok(name);
81            }
82        }
83        Err("Failed to generate unique name".to_string())
84    }
85
86    pub fn recover_crash(&self) -> Result<(), String> {
87        let _guard = self.lock.lock().map_err(|e| e.to_string())?;
88        let mut map = self.load()?;
89        let mut chg = false;
90        for s in map.values_mut() {
91            if !s.name.starts_with("ia-") && s.status == "running" {
92                s.status = "idle".to_string();
93                chg = true;
94            }
95        }
96        if chg { self.save(&map)?; }
97        Ok(())
98    }
99
100    pub fn create_session(&self, chat_id: i64, status: &str, last_prompt: &str) -> Result<NamedSession, String> {
101        let _guard = self.lock.lock().map_err(|e| e.to_string())?;
102        let mut map = self.load()?;
103        let count = map.values().filter(|s| s.chat_id == chat_id).count();
104        if count >= 10 {
105            return Err("Max 10 named sessions per chat allowed".to_string());
106        }
107        let name = self.generate_unique_name(&map)?;
108        let mut prompt = last_prompt.to_string();
109        if status == "running" && prompt.len() > 4000 {
110            prompt = prompt.chars().take(4000).collect();
111        }
112        let ns = NamedSession {
113            name: name.clone(),
114            chat_id,
115            status: status.to_string(),
116            last_prompt: prompt,
117        };
118        map.insert(name, ns.clone());
119        self.save(&map)?;
120        Ok(ns)
121    }
122
123    pub fn update_session_status(&self, name: &str, status: &str, last_prompt: Option<&str>) -> Result<NamedSession, String> {
124        let _guard = self.lock.lock().map_err(|e| e.to_string())?;
125        let mut map = self.load()?;
126        if let Some(s) = map.get_mut(name) {
127            s.status = status.to_string();
128            if let Some(p) = last_prompt {
129                s.last_prompt = p.to_string();
130            }
131            if s.status == "running" && s.last_prompt.len() > 4000 {
132                s.last_prompt = s.last_prompt.chars().take(4000).collect();
133            }
134            let cloned = s.clone();
135            self.save(&map)?;
136            Ok(cloned)
137        } else {
138            Err(format!("Named session not found: {}", name))
139        }
140    }
141
142    pub fn get_session(&self, name: &str) -> Result<Option<NamedSession>, String> {
143        let _guard = self.lock.lock().map_err(|e| e.to_string())?;
144        let map = self.load()?;
145        Ok(map.get(name).cloned())
146    }
