1//! # Session Data Structure Models
2//!
3//! ## Overview
4//! Defines the [`SessionData`] struct capturing runtime state, provider, models, and history counts
5//! for active messaging streams.
6//!
7//! ## Collaboration Graph
8//! - Persisted by [`SessionManager`](super::manager::SessionManager).
9//! - Loaded/updated dynamically as CLI streams write output chunks.

11//! ## Search Tags
12//! #session-data, #model-override, #session-history, #metadata

66#[derive(Clone, Debug, Serialize, Deserialize)]
67pub struct SessionData {
68    #[serde(default = "default_transport")]
69    pub transport: String,
70    pub chat_id: i64,
71    pub topic_id: Option<i64>,
72    pub topic_name: Option<String>,
73    #[serde(default = "default_provider")]
74    pub provider: String,
75    #[serde(default = "default_model")]
76    pub model: String,
77    #[serde(default = "default_now_iso")]
78    pub created_at: String,
79    #[serde(default = "default_now_iso")]
80    pub last_active: String,
81    #[serde(default)]
82    pub provider_sessions: HashMap<String, ProviderSessionData>,
83    #[serde(default)]
84    pub language: Option<String>,
85    #[serde(default)]
86    pub last_progress_msg_id: Option<i32>,
87    #[serde(default)]
88    pub pending_attachments: Vec<String>,
89
90    // Legacy fields used for migration
91    #[serde(skip_serializing, default)]
92    pub session_id: Option<String>,
93    #[serde(skip_serializing, default)]
94    pub message_count: Option<i64>,
95    #[serde(skip_serializing, default)]
96    pub total_cost_usd: Option<f64>,
97    #[serde(skip_serializing, default)]
98    pub total_tokens: Option<i64>,
99}
100
101fn default_transport() -> String { "tg".to_string() }
102fn default_provider() -> String { "claude".to_string() }
103fn default_model() -> String { "opus".to_string() }
104fn default_now_iso() -> String { Utc::now().to_rfc3339() }
105
106impl SessionData {
107    pub fn new(chat_id: i64, transport: String, topic_id: Option<i64>, provider: String, model: String) -> Self {
108        let now = default_now_iso();
109        Self {
110            transport,
111            chat_id,
112            topic_id,
113            topic_name: None,
114            provider,
115            model,
116            created_at: now.clone(),
117            last_active: now,
118            provider_sessions: HashMap::new(),
119            language: None,
120            last_progress_msg_id: None,
121            pending_attachments: Vec::new(),
122            session_id: None,
123            message_count: None,
124            total_cost_usd: None,
125            total_tokens: None,
126        }
127    }
128
129    pub fn session_key(&self) -> SessionKey {
130        SessionKey {
131            transport: self.transport.clone(),
132            chat_id: self.chat_id,
133            topic_id: self.topic_id,
134        }
135    }
136
137    pub fn get_session_id(&self, provider: &str) -> String {
138        self.provider_sessions.get(provider)
139            .map(|ps| ps.session_id.clone())
140            .unwrap_or_default()
141    }
142
143    pub fn set_session_id(&mut self, provider: &str, session_id: &str) {
144        let ps = self.provider_sessions.entry(provider.to_string()).or_default();
145        ps.session_id = session_id.to_string();
146    }
147
148    pub fn migrate_legacy_metrics(&mut self) {
149        let has_legacy = self.session_id.is_some()
150            || self.message_count.is_some()
151            || self.total_cost_usd.is_some()
152            || self.total_tokens.is_some();
153
154        if has_legacy && !self.provider.is_empty() {
155            let ps = self.provider_sessions.entry(self.provider.clone()).or_default();
156            if let Some(ref sid) = self.session_id {
157                ps.session_id = sid.clone();
158            }
159            if let Some(cnt) = self.message_count {
160                ps.message_count = cnt;
161            }
162            if let Some(cost) = self.total_cost_usd {
163                ps.total_cost_usd = cost;
164            }
165            if let Some(tok) = self.total_tokens {
166                ps.total_tokens = tok;
167            }
168        }
169    }
170
171    pub fn clear_provider_session(&mut self, provider: &str) {
172        if let Some(ps) = self.provider_sessions.get_mut(provider) {
173            ps.session_id = String::new();
174        }
175    }
176}
