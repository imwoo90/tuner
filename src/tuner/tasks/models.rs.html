11/// Allowed task priority levels
12pub const TASK_PRIORITIES: &[&str] = &["interactive", "background", "batch"];
13/// Default task priority level
14pub const DEFAULT_PRIORITY: &str = "background";
15
16/// Configuration settings for the task system
17#[derive(Clone, Debug)]
18pub struct TasksConfig {
19    pub enabled: bool,
20    pub max_parallel: usize,
21    pub timeout_seconds: f64,
22}
23
24impl Default for TasksConfig {
25    fn default() -> Self {
26        Self {
27            enabled: true,
28            max_parallel: 5,
29            timeout_seconds: 60.0,
30        }
31    }
32}

34/// Coerces a string slice to a valid task priority, or defaults to "background"
35pub fn normalise_priority(value: Option<&str>) -> String {
36    match value {
37        Some(p) if TASK_PRIORITIES.contains(&p) => p.to_string(),
38        _ => DEFAULT_PRIORITY.to_string(),
39    }
40}

54/// A persisted task entry in the registry
55#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
56pub struct TaskEntry {
57    pub task_id: String,
58    pub chat_id: i64,
59    #[serde(default = "default_parent_agent")]
60    pub parent_agent: String,
61    #[serde(default)]
62    pub name: String,
63    #[serde(default)]
64    pub prompt_preview: String,
65    #[serde(default)]
66    pub provider: String,
67    #[serde(default)]
68    pub model: String,
69    #[serde(default = "default_status")]
70    pub status: String, // "running", "done", "failed", "cancelled", "waiting"
71    #[serde(default)]
72    pub session_id: String,
73    #[serde(default)]
74    pub created_at: f64,
75    #[serde(default)]
76    pub completed_at: f64,
77    #[serde(default)]
78    pub elapsed_seconds: f64,
79    #[serde(default)]
80    pub error: String,
81    #[serde(default)]
82    pub result_preview: String,
83    #[serde(default)]
84    pub question_count: i64,
85    #[serde(default)]
86    pub num_turns: i64,
87    #[serde(default)]
88    pub last_question: String,
89    #[serde(default)]
90    pub original_prompt: String,
91    #[serde(default)]
92    pub thinking: String,
93    #[serde(default)]
94    pub tasks_dir: String,
95    #[serde(default, skip_serializing_if = "Option::is_none")]
96    pub thread_id: Option<i64>,
97    #[serde(default = "default_priority", deserialize_with = "deserialize_priority")]
98    pub priority: String,
99    #[serde(default)]
100    pub depends_on: Vec<String>,
101}
102
103fn deserialize_priority<'de, D>(deserializer: D) -> Result<String, D::Error>
104where
105    D: serde::Deserializer<'de>,
106{
107    let opt: Option<String> = Deserialize::deserialize(deserializer)?;
108    Ok(normalise_priority(opt.as_deref()))
109}
