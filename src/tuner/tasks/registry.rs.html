33fn seed_folder_helper(resolved_dir: &Path, task_id: &str, entry: &TaskEntry, prompt: &str) -> Result<()> {
34    let folder = resolved_dir.join(task_id);
35    fs::create_dir_all(&folder)?;
36    seed_task_folder(&folder, entry, prompt)?;
37    Ok(())
38}

40fn get_adj(entries: &HashMap<String, TaskEntry>, task_id: String, submit: &TaskSubmit) -> HashMap<String, Vec<String>> {
41    let mut adj = HashMap::new();
42    for (tid, entry) in entries.iter() {
43        adj.insert(tid.clone(), entry.depends_on.clone());
44    }
45    adj.insert(task_id, submit.depends_on.clone());
46    adj
47}

49fn make_entry(
50    task_id: String,
51    submit: &TaskSubmit,
52    provider: String,
53    model: String,
54    thinking: String,
55    resolved_dir: &Path,
56    priority: String,
57) -> TaskEntry {
58    let name = if submit.name.trim().is_empty() { task_id.clone() } else { submit.name.clone() };
59    let preview: String = submit.prompt.chars().take(PROMPT_PREVIEW_LEN).collect();
60    TaskEntry {
61        task_id,
62        chat_id: submit.chat_id,
63        parent_agent: submit.parent_agent.clone(),
64        name,
65        prompt_preview: preview,
66        provider,
67        model,
68        status: "running".to_string(),
69        session_id: String::new(),
70        created_at: chrono::Utc::now().timestamp() as f64,
71        completed_at: 0.0,
72        elapsed_seconds: 0.0,
73        error: String::new(),
74        result_preview: String::new(),
75        question_count: 0,
76        num_turns: 0,
77        last_question: String::new(),
78        original_prompt: submit.prompt.clone(),
79        thinking,
80        tasks_dir: resolved_dir.to_string_lossy().to_string(),
81        thread_id: submit.thread_id,
82        priority,
83        depends_on: submit.depends_on.clone(),
84    }
85}

87impl TaskRegistry {
88    /// Creates or loads a TaskRegistry from a JSON file.
89    pub fn new(registry_path: PathBuf, default_tasks_dir: PathBuf) -> Result<Self> {
90        let registry = Self {
91            registry_path,
92            default_tasks_dir,
93            entries: Mutex::new(HashMap::new()),
94        };
95        registry.load()?;
96        registry.cleanup_orphans()?;
97        Ok(registry)
98    }
99
100    fn load(&self) -> Result<()> {
101        if !self.registry_path.exists() {
102            return Ok(());
103        }
104        let content = fs::read_to_string(&self.registry_path)
105            .context("Failed to read task registry file")?;
106        if content.trim().is_empty() {
107            return Ok(());
108        }
109        let data: RegistryData = serde_json::from_str(&content)
110            .context("Failed to deserialize task registry")?;
111
112        let mut entries = self.entries.lock().unwrap();
113        for mut entry in data.tasks {
114            if entry.status == "running" {
115                entry.status = "failed".to_string();
116                entry.error = "Bot restarted while task was running".to_string();
117            }
118            entries.insert(entry.task_id.clone(), entry);
119        }
120        Ok(())
121    }
122
123    pub(crate) fn persist(&self, entries: &HashMap<String, TaskEntry>) -> Result<()> {
124        let data = RegistryData {
125            tasks: entries.values().cloned().collect(),
126        };
127        let temp = self.registry_path.with_extension("tmp");
128        let content = serde_json::to_string_pretty(&data)
129            .context("Failed to serialize task registry")?;
130
131        if let Some(parent) = self.registry_path.parent() {
132            fs::create_dir_all(parent)?;
133        }
134        fs::write(&temp, content)?;
135        fs::rename(&temp, &self.registry_path)?;
136        Ok(())
137    }
138
139    /// Creates a new task entry in the registry and seeds its task folder.
140    pub fn create(
141        &self,
142        submit: TaskSubmit,
143        provider: String,
144        model: String,
145        thinking: String,
146        tasks_dir_override: Option<PathBuf>,
147        priority_override: Option<String>,
148    ) -> Result<TaskEntry> {
149        let mut entries = self.entries.lock().unwrap();
150
151        let task_id = loop {
152            let bytes: [u8; 4] = rand::random();
153            let candidate = hex::encode(bytes);
154            if !entries.contains_key(&candidate) {
155                break candidate;
156            }
157        };
158
159        let adj = get_adj(&entries, task_id.clone(), &submit);
160        if check_cycle(&adj) {
161            return Err(anyhow!("Cycle detected in task dependencies"));
162        }
163
164        let resolved_dir = tasks_dir_override.unwrap_or_else(|| self.default_tasks_dir.clone());
165        let priority = normalise_priority(priority_override.as_deref().or(Some(&submit.priority)));
166
167        let entry = make_entry(task_id.clone(), &submit, provider, model, thinking, &resolved_dir, priority);
168
169        entries.insert(task_id.clone(), entry.clone());
170        self.persist(&entries)?;
171
172        seed_folder_helper(&resolved_dir, &task_id, &entry, &submit.prompt)?;
173
174        Ok(entry)
175    }

177    /// Retrieve a task entry by its ID.
178    pub fn get(&self, task_id: &str) -> Option<TaskEntry> {
179        let entries = self.entries.lock().unwrap();
180        entries.get(task_id).cloned()
181    }
182
183    /// Find a task by name within a chat.
184    pub fn find_by_name(&self, chat_id: i64, name: &str) -> Option<TaskEntry> {
185        let entries = self.entries.lock().unwrap();
186        let lower = name.to_lowercase();
187        for entry in entries.values() {
188            if entry.chat_id == chat_id && entry.name.to_lowercase() == lower {
189                return Some(entry.clone());
190            }
191        }
192        None
193    }
194
195    /// List active running tasks.
196    pub fn list_active(&self, chat_id: Option<i64>) -> Vec<TaskEntry> {
197        let entries = self.entries.lock().unwrap();
198        let mut active: Vec<TaskEntry> = entries
199            .values()
200            .filter(|e| e.status == "running" && chat_id.map_or(true, |c| e.chat_id == c))
201            .cloned()
202            .collect();
203        active.sort_by(|a, b| a.created_at.partial_cmp(&b.created_at).unwrap());
204        active
205    }
206
207    /// List all tasks.
208    pub fn list_all(&self, chat_id: Option<i64>, parent_agent: Option<&str>) -> Vec<TaskEntry> {
209        let entries = self.entries.lock().unwrap();
210        let mut all: Vec<TaskEntry> = entries
211            .values()
212            .filter(|e| {
213                chat_id.map_or(true, |c| e.chat_id == c)
214                    && parent_agent.map_or(true, |pa| e.parent_agent == pa)
215            })
216            .cloned()
217            .collect();
218        all.sort_by(|a, b| b.created_at.partial_cmp(&a.created_at).unwrap());
219        all
220    }

222    /// Update status and optional fields.
223    pub fn update_status<F>(&self, task_id: &str, update_fn: F) -> Result<bool>
224    where
225        F: FnOnce(&mut TaskEntry),
226    {
227        let mut entries = self.entries.lock().unwrap();
228        if let Some(entry) = entries.get_mut(task_id) {
229            update_fn(entry);
230            self.persist(&entries)?;
231            Ok(true)
232        } else {
233            Ok(false)
234        }
235    }
236
237    /// Returns the task's workspace folder.
238    pub fn task_folder(&self, task_id: &str) -> PathBuf {
239        let entries = self.entries.lock().unwrap();
240        if let Some(entry) = entries.get(task_id) {
241            self.task_folder_internal(entry)
242        } else {
243            self.default_tasks_dir.join(task_id)
244        }
245    }
246
247    pub(crate) fn task_folder_internal(&self, entry: &TaskEntry) -> PathBuf {
248        if !entry.tasks_dir.is_empty() {
249            PathBuf::from(&entry.tasks_dir).join(&entry.task_id)
250        } else {
251            self.default_tasks_dir.join(&entry.task_id)
252        }
253    }
254
255    /// Returns the path to TASKMEMORY.md for a given task.
256    pub fn taskmemory_path(&self, task_id: &str) -> PathBuf {
257        self.task_folder(task_id).join("TASKMEMORY.md")
258    }

293fn seed_task_folder(folder: &Path, entry: &TaskEntry, prompt: &str) -> Result<()> {
294    let taskmemory_path = folder.join("TASKMEMORY.md");
295    if !taskmemory_path.exists() {
296        let created = chrono::Local::now().format("%Y-%m-%d %H:%M:%S").to_string();
297        let prompt_part: String = prompt.chars().take(500).collect();
298        let initial_memory = format!(
299            "# Task: {}\n\nCreated: {}\nProvider: {}/{}\n\n## Task Description\n\n{}\n\n## Progress\n\n_Update this section as you work._\n",
300            entry.name, created, entry.provider, entry.model, prompt_part
301        );
302        fs::write(&taskmemory_path, initial_memory)?;
303    }
304
305    let rules_content = TASK_RULES.replace("{taskmemory_path}", &taskmemory_path.to_string_lossy());
306    for name in &["CLAUDE.md", "AGENTS.md", "GEMINI.md"] {
307        let rules_path = folder.join(name);
308        fs::write(rules_path, &rules_content)?;
309    }
310    Ok(())
