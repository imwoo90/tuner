12use std::sync::OnceLock;
13use regex::{Regex, RegexBuilder};
14
15fn get_patterns() -> &'static [(Regex, &'static str)] {
16    static PATTERNS: OnceLock<Vec<(Regex, &'static str)>> = OnceLock::new();
17    PATTERNS.get_or_init(|| {
18        let raw = vec![
19            (r"ignore\s+(all\s+)?(previous|prior|above)\s+(instructions?|prompts?)", "instruction_override"),
20            (r"disregard\s+(all\s+)?(previous|prior|above)", "instruction_override"),
21            (r"forget\s+(everything|all|your)\s+(instructions?|rules?)", "instruction_override"),
22            (r"you\s+are\s+now\s+(a|an)\s+", "role_hijack"),
23            (r"new\s+instructions?:", "role_hijack"),
24            (r"system\s*:\s*prompt", "fake_system_prompt"),
25            (r"<\|(?:im_start|im_end|system|endoftext)\|>", "special_token"),
26            (r"\[INST\]|\[/INST\]|<<SYS>>|<</SYS>>", "llama_markers"),
27            (r"(?:^|\n)\s*(?:Human|Assistant|System)\s*:", "anthropic_markers"),
28            (r"GROUND_RULES|(?:AGENT_)?SOUL\.md|(?:AGENT_)?SYSTEM\.md|BOOTSTRAP\.md|(?:AGENT_)?IDENTITY\.md", "internal_file_ref"),
29            (r"mem_add\.py|mem_edit\.py|mem_delete\.py|task_add\.py", "tool_injection"),
30            (r"--system-prompt|--append-system-prompt|--permission-mode", "cli_flag_injection"),
31            (r"<file:[^>]+>", "file_tag_injection"),
32        ];
33        raw.into_iter()
34            .map(|(p, cat)| {
35                let re = RegexBuilder::new(p)
36                    .case_insensitive(true)
37                    .build()
38                    .unwrap();
39                (re, cat)
40            })
41            .collect()
42    })
43}

45/// Scan text for prompt injection patterns. Empty vector = clean.
46pub fn detect_suspicious_patterns(text: &str) -> Vec<String> {
47    let folded = fold_fullwidth(text);
48    let mut matched = Vec::new();
49    for (re, cat) in get_patterns() {
50        if re.is_match(&folded) {
51            matched.push((*cat).to_string());
52        }
53    }
54    matched
55}

57/// Helper to fold fullwidth Unicode characters to standard ASCII equivalents.
58pub fn fold_fullwidth(text: &str) -> String {
59    text.chars().map(fold_fullwidth_char).collect()
60}
61
62/// Helper to fold a single character if it is a fullwidth equivalent.
63pub fn fold_fullwidth_char(c: char) -> char {
64    let code = c as u32;
65    if (0xFF21..=0xFF3A).contains(&code) || (0xFF41..=0xFF5A).contains(&code) {
66        char::from_u32(code - 0xFEE0).unwrap_or(c)
67    } else if code == 0xFF1C {
68        '<'
69    } else if code == 0xFF1E {
70        '>'
71    } else {
72        c
73    }
74}
