1//! # Cron Engine Event Loop
2//!
3//! ## Overview
4//! Continuously polls active cron jobs, evaluates timezone-adjusted upcoming runs, checks quiet hours,
5//! and triggers background agent runs via [`AntigravityCli`].
6//!
7//! ## Collaboration Graph
8//! - Ticked by [`CronScheduler::start`] event loop spawning background monitoring threads.
9//! - Queries [`super::manager::CronManager`] to retrieve active [`CronJob`] structures.
10//! - Utilizes [`AntigravityCli`] to run prompt instructions.
11//! - Dispatches results as envelopes to the central [`MessageBus`](crate::bus::bus::MessageBus).
12//!
13//! ## Key Types
14//! - [`CronScheduler`]: Controller holding configuration, client bindings, message bus adapters, and quiet-hour status.
15//!
16//! ## Search Tags
17//! #cron-engine, #tick-loop, #quiet-hours, #task-spawner, #time-evaluator

49    fn normalize_cron_expression(expr: &str) -> String {
50        let parts: Vec<&str> = expr.split_whitespace().collect();
51        if parts.len() == 5 {
52            format!("0 {}", expr)
53        } else {
54            expr.to_string()
55        }
56    }
57
58    pub(crate) fn calculate_next_run(&self, job: &CronJob) -> Result<DateTime<Utc>, String> {
59        let normalized = Self::normalize_cron_expression(&job.schedule);
60        let sched = Schedule::from_str(&normalized).map_err(|e| e.to_string())?;
61        let tz_str = if job.timezone.is_empty() { "UTC" } else { &job.timezone };
62        let tz: Tz = tz_str.parse().map_err(|e| format!("Invalid timezone: {}", e))?;
63        let next_local = sched.upcoming(tz).next().ok_or_else(|| "No upcoming execution time found".to_string())?;
64        Ok(next_local.with_timezone(&Utc))
65    }

67    pub fn start(self: Arc<Self>) {
68        let scheduler = self.clone();
69        tokio::spawn(async move {
70            let mut last_mtime: Option<SystemTime> = None;
71            let mut next_run_times: HashMap<String, DateTime<Utc>> = HashMap::new();
72            let mut check_interval = tokio::time::interval(tokio::time::Duration::from_secs(5));
73            println!("🤖 [tuner] CronScheduler started");
74            loop {
75                check_interval.tick().await;
76                let _ = scheduler.tick(&mut last_mtime, &mut next_run_times).await;
77            }
78        });
79    }

81    async fn tick(
82        &self,
83        last_mtime: &mut Option<SystemTime>,
84        next_run_times: &mut HashMap<String, DateTime<Utc>>,
85    ) -> Result<(), String> {
86        let _ = self.reload_if_changed(last_mtime, next_run_times).await;
87        let _ = self.run_due_jobs(next_run_times).await;
88        Ok(())
89    }
90
91    async fn reload_if_changed(
92        &self,
93        last_mtime: &mut Option<SystemTime>,
94        next_run_times: &mut HashMap<String, DateTime<Utc>>,
95    ) -> Result<(), String> {
96        let current_mtime = std::fs::metadata(self.manager.jobs_path()).ok().and_then(|m| m.modified().ok());
97        if current_mtime != *last_mtime {
98            *last_mtime = current_mtime;
99            println!("🤖 [tuner] CronScheduler: reloading jobs");
100            if let Ok(jobs) = self.manager.list_jobs().await {
101                let mut updated = HashMap::new();
102                for job in jobs {
103                    if job.enabled {
104                        if let Ok(next) = self.calculate_next_run(&job) {
105                            let next_run = next_run_times.get(&job.id).filter(|&&e| e > Utc::now()).copied().unwrap_or(next);
106                            updated.insert(job.id, next_run);
107                        }
108                    }
109                }
110                *next_run_times = updated;
111            }
112        }
113        Ok(())
114    }
115
116    async fn run_due_jobs(&self, next_run_times: &mut HashMap<String, DateTime<Utc>>) -> Result<(), String> {
117        let now = Utc::now();
118        let mut jobs_to_run = Vec::new();
119        for (job_id, next_run) in next_run_times.iter() {
120            if now >= *next_run { jobs_to_run.push(job_id.clone()); }
121        }
122        for job_id in jobs_to_run {
123            if let Ok(Some(job)) = self.manager.get_job(&job_id).await {
124                let self_clone = Arc::new(Self::new(self.config.clone(), self.manager.clone(), self.cli.clone(), self.bus.clone()));
125                let job_for_task = job.clone();
126                tokio::spawn(async move { let _ = self_clone.execute_job(job_for_task).await; });
127                if let Ok(next_run) = self.calculate_next_run(&job) {
128                    next_run_times.insert(job_id, next_run);
129                } else {
130                    next_run_times.remove(&job_id);
131                }
132            }
133        }
134        Ok(())
135    }

137    fn check_quiet_hours(&self, job: &CronJob) -> bool {

139    }

141    pub fn check_quiet_hours_at(&self, job: &CronJob, now: DateTime<Utc>) -> bool {
142        if let (Some(qs), Some(qe)) = (job.quiet_start, job.quiet_end) {
143            let tz_str = if job.timezone.is_empty() { "UTC" } else { &job.timezone };
144            if let Ok(tz) = tz_str.parse::<Tz>() {
145                let now_local = now.with_timezone(&tz);
146                let start = NaiveTime::from_hms_opt(qs, 0, 0).unwrap();
147                let end = NaiveTime::from_hms_opt(qe, 0, 0).unwrap();
148                return crate::heartbeat::quiet::is_within_quiet_hours(&now_local, start, end);
149            }
150        }
151        false
152    }

167    async fn run_cli_command(&self, job: &CronJob, workspace: std::path::PathBuf, cli: &AntigravityCli) -> Result<(), String> {
168        let job_id = job.id.clone();
169        let job_title = job.title.clone();
170        let chat_id = job.chat_id;
171        let topic_id = job.topic_id;
172        let enriched = self.get_enriched_prompt(&job.agent_instruction, &job.task_folder);
173        let res = cli.send(&enriched, None, false, workspace).await;
174        match res {
175            Ok(resp) => {
176                let status = if resp.is_error {
177                    format!("error:exit_{}", resp.returncode.unwrap_or(1))
178                } else {
179                    "success".to_string()
180                };
181                self.manager.update_run_status(&job_id, &status).await?;
182                if !job.silent_on_success || resp.is_error {
183                    let mut env = crate::bus::adapters::from_cron_result(
184                        &job_title,
185                        &resp.result,
186                        &status,
187                        Some(chat_id),
188                        topic_id,
189                        Some(&self.config.transport),
190                    );
191                    self.bus.submit(&mut env).await;
192                }
193            }
194            Err(e) => {
195                self.manager.update_run_status(&job_id, &format!("error:{}", e)).await?;
196                let mut env = crate::bus::adapters::from_cron_result(
197                    &job_title,
198                    "",
199                    &format!("error:{}", e),
200                    Some(chat_id),
201                    topic_id,
202                    Some(&self.config.transport),
203                );
204                self.bus.submit(&mut env).await;
205            }
206        }
207        Ok(())
208    }

210    async fn execute_job(&self, job: CronJob) -> Result<(), String> {
211        if self.check_quiet_hours(&job) {
212            println!("🤖 [tuner] Cron job {} skipped due to quiet hours", job.title);
213            return Ok(());
214        }
215        println!("🤖 [tuner] Cron job executing: {}", job.title);
216        let ctr = self.config.working_dir.join("cron_tasks");
217        let workspace = ctr.join(&job.task_folder);
218        if !crate::security::paths::is_path_safe(&workspace, &[ctr]) {
219            let _ = self.manager.update_run_status(&job.id, "error:path_outside_allowed_roots").await;
220            return Err("Cron task folder outside allowed roots".into());
221        }
222        if !workspace.is_dir() {
223            let err_msg = format!("Cron task folder missing: {}", workspace.display());
224            eprintln!("❌ [tuner] {}", err_msg);
225            self.manager.update_run_status(&job.id, "error:folder_missing").await?;
226            return Err(err_msg);
227        }
228
229        let mut jc = (*self.config).clone();
230        jc.chat_id = job.chat_id;
231        jc.topic_id = job.topic_id;
232        jc.transport = job.transport.clone();
233        if let Some(m) = &job.model { jc.model = Some(m.clone()); }
234        if let Some(p) = &job.provider { jc.provider = p.clone(); }
235        if !job.cli_parameters.is_empty() {
236            jc.cli_parameters.insert("antigravity".to_string(), job.cli_parameters.clone());
237        }
238        if let Some(r) = &job.reasoning_effort {
239            jc.cli_parameters.entry("antigravity".to_string()).or_default()
240                .extend(vec!["-c".to_string(), format!("model_reasoning_effort={}", r)]);
241        }
242
243        let job_cli = AntigravityCli::new(jc);
244        self.run_cli_command(&job, workspace, &job_cli).await
245    }
