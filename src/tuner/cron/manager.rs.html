1//! # Cron Task Registry and State Manager
2//!
3//! ## Overview
4//! Handles loading, saving, registering, and tracking execution states of [`CronJob`] definitions.
5//! Employs asynchronous mutex locks to prevent race conditions during JSON file persistence.
6//!
7//! ## Collaboration Graph
8//! - Reads/writes local json files representing scheduled configurations.
9//! - Consulted by [`super::scheduler::CronScheduler`] to reload jobs when changes are detected on disk.
10//!
11//! ## Key Structures
12//! - [`CronJob`]: State representation containing schedule, provider, timezone, models, and execution logs.
13//! - [`CronManager`]: Service coordinator handling persistence and access locks.
14//!
15//! ## Search Tags
16//! #cron-registry, #job-persistence, #state-lock, #cron-jobs
17

102#[derive(Serialize, Deserialize)]
103struct CronJobsFile {
104    pub jobs: Vec<CronJob>,
105}
106
107pub struct CronManager {
108    jobs_path: PathBuf,
109    lock: tokio::sync::Mutex<()>,
110}
111
112impl CronManager {
113    pub fn new(jobs_path: PathBuf) -> Self {
114        Self {
115            jobs_path,
116            lock: tokio::sync::Mutex::new(()),
117        }
118    }
119
120    pub fn jobs_path(&self) -> &PathBuf {
121        &self.jobs_path
122    }
123
124    pub fn load(&self) -> Result<Vec<CronJob>, String> {
125        if !self.jobs_path.exists() {
126            return Ok(Vec::new());
127        }
128        let content = fs::read_to_string(&self.jobs_path).map_err(|e| e.to_string())?;
129        if content.trim().is_empty() {
130            return Ok(Vec::new());
131        }
132        let file_data: CronJobsFile = serde_json::from_str(&content).map_err(|e| e.to_string())?;
133        Ok(file_data.jobs)
134    }
135
136    pub fn save(&self, jobs: &[CronJob]) -> Result<(), String> {
137        let temp_path = self.jobs_path.with_extension("tmp");
138        let file_data = CronJobsFile { jobs: jobs.to_vec() };
139        let content = serde_json::to_string_pretty(&file_data).map_err(|e| e.to_string())?;
140        if let Some(parent) = self.jobs_path.parent() {
141            fs::create_dir_all(parent).map_err(|e| e.to_string())?;
142        }
143        fs::write(&temp_path, content).map_err(|e| e.to_string())?;
144        fs::rename(&temp_path, &self.jobs_path).map_err(|e| e.to_string())?;
145        Ok(())
146    }
147
148    pub async fn add_job(&self, job: CronJob) -> Result<(), String> {
149        let _guard = self.lock.lock().await;
150        let mut jobs = self.load()?;
151        if jobs.iter().any(|j| j.id == job.id) {
152            return Err(format!("Job '{}' already exists", job.id));
153        }
154        jobs.push(job);
155        self.save(&jobs)?;
156        Ok(())
157    }
158
159    pub async fn remove_job(&self, job_id: &str) -> Result<bool, String> {
160        let _guard = self.lock.lock().await;
161        let mut jobs = self.load()?;
162        let before_len = jobs.len();
163        jobs.retain(|j| j.id != job_id);
164        let removed = jobs.len() < before_len;
165        if removed {
166            self.save(&jobs)?;
167        }
168        Ok(removed)
169    }
170
171    pub async fn list_jobs(&self) -> Result<Vec<CronJob>, String> {
172        let _guard = self.lock.lock().await;
173        self.load()
174    }
175
176    pub async fn get_job(&self, job_id: &str) -> Result<Option<CronJob>, String> {
177        let _guard = self.lock.lock().await;
178        let jobs = self.load()?;
179        Ok(jobs.into_iter().find(|j| j.id == job_id))
180    }
181
182    pub async fn set_enabled(&self, job_id: &str, enabled: bool) -> Result<bool, String> {
183        let _guard = self.lock.lock().await;
184        let mut jobs = self.load()?;
185        let mut changed = false;
186        if let Some(job) = jobs.iter_mut().find(|j| j.id == job_id) {
187            if job.enabled != enabled {
188                job.enabled = enabled;
189                changed = true;
190            }
191        }
192        if changed {
193            self.save(&jobs)?;
194        }
195        Ok(changed)
196    }
197
198    pub async fn set_all_enabled(&self, enabled: bool) -> Result<usize, String> {
199        let _guard = self.lock.lock().await;
200        let mut jobs = self.load()?;
201        let mut changed_count = 0;
202        for job in jobs.iter_mut() {
203            if job.enabled != enabled {
204                job.enabled = enabled;
205                changed_count += 1;
206            }
207        }
208        if changed_count > 0 {
209            self.save(&jobs)?;
210        }
211        Ok(changed_count)
212    }
213
214    pub async fn update_run_status(&self, job_id: &str, status: &str) -> Result<(), String> {
215        let _guard = self.lock.lock().await;
216        let mut jobs = self.load()?;
217        if let Some(job) = jobs.iter_mut().find(|j| j.id == job_id) {
218            job.last_run_at = Some(Utc::now().to_rfc3339());
219            job.last_run_status = Some(status.to_string());
220            self.save(&jobs)?;
221        }
222        Ok(())
223    }
224}
