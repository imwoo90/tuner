13fn get_mtime(p: &Path) -> Option<chrono::DateTime<chrono::Utc>> {
14    std::fs::metadata(p).and_then(|m| m.modified()).map(chrono::DateTime::from).ok()
15}

17pub fn check_claude(paths: &DuctorPaths) -> AuthResult {
18    let home = get_home_dir(paths);
19    let claude_dir = home.join(".claude");
20    let credentials = claude_dir.join(".credentials.json");
21
22    if credentials.is_file() {
23        return AuthResult {
24            provider: "claude".to_string(),
25            status: AuthStatus::Authenticated,
26            auth_file: Some(credentials.clone()),
27            auth_age: get_mtime(&credentials),
28        };
29    }
30    if std::env::var("ANTHROPIC_API_KEY").map(|s| !s.trim().is_empty()).unwrap_or(false) {
31        return AuthResult {
32            provider: "claude".to_string(),
33            status: AuthStatus::Authenticated,
34            auth_file: None,
35            auth_age: None,
36        };
37    }
38    if check_claude_cli_logged_in() {
39        return AuthResult {
40            provider: "claude".to_string(),
41            status: AuthStatus::Authenticated,
42            auth_file: None,
43            auth_age: None,
44        };
45    }
46    let status = if claude_dir.is_dir() { AuthStatus::Installed } else { AuthStatus::NotFound };
47    AuthResult { provider: "claude".to_string(), status, auth_file: None, auth_age: None }
48}

50pub fn check_codex(paths: &DuctorPaths) -> AuthResult {
51    let home = get_home_dir(paths);
52    let codex_home = std::env::var("CODEX_HOME")
53        .map(PathBuf::from)
54        .unwrap_or_else(|_| home.join(".codex"));
55    let auth_file = codex_home.join("auth.json");
56
57    if auth_file.is_file() {
58        return AuthResult {
59            provider: "codex".to_string(),
60            status: AuthStatus::Authenticated,
61            auth_file: Some(auth_file.clone()),
62            auth_age: get_mtime(&auth_file),
63        };
64    }
65    if std::env::var("OPENAI_API_KEY").map(|s| !s.trim().is_empty()).unwrap_or(false) {
66        return AuthResult {
67            provider: "codex".to_string(),
68            status: AuthStatus::Authenticated,
69            auth_file: None,
70            auth_age: None,
71        };
72    }
73    let status = if codex_home.join("version.json").is_file() || codex_home.join("config.toml").is_file() {
74        AuthStatus::Installed
75    } else {
76        AuthStatus::NotFound
77    };
78    AuthResult { provider: "codex".to_string(), status, auth_file: None, auth_age: None }
79}

81pub fn check_gemini(paths: &DuctorPaths) -> AuthResult {
82    if !find_gemini_cli() {
83        return AuthResult { provider: "gemini".to_string(), status: AuthStatus::NotFound, auth_file: None, auth_age: None };
84    }
85    let home = get_home_dir(paths);
86    let gemini_home = std::env::var("GEMINI_CLI_HOME")
87        .map(|s| PathBuf::from(s).join(".gemini"))
88        .unwrap_or_else(|_| home.join(".gemini"));
89
90    let oauth_file = gemini_home.join("oauth_creds.json");
91    if oauth_file.is_file() && std::fs::metadata(&oauth_file).map(|m| m.len() > 0).unwrap_or(false) {
92        return AuthResult { provider: "gemini".to_string(), status: AuthStatus::Authenticated, auth_file: Some(oauth_file.clone()), auth_age: get_mtime(&oauth_file) };
93    }
94    let env_path = gemini_home.join(".env");
95    let parent_env_path = gemini_home.parent().map(|p| p.join(".env")).unwrap_or_else(|| PathBuf::from("/nonexistent"));
96    let has_env = std::env::var("GEMINI_API_KEY").map(|s| !s.trim().is_empty()).unwrap_or(false)
97        || std::env::var("GOOGLE_API_KEY").map(|s| !s.trim().is_empty()).unwrap_or(false)
98        || (std::env::var("GOOGLE_CLOUD_PROJECT").is_ok() && std::env::var("GOOGLE_CLOUD_LOCATION").is_ok());
99    
100    if has_env || read_dotenv_has_gemini_keys(&env_path) || read_dotenv_has_gemini_keys(&parent_env_path) || config_has_gemini_key(paths) {
101        return AuthResult { provider: "gemini".to_string(), status: AuthStatus::Authenticated, auth_file: None, auth_age: None };
102    }
103    if let Some(status) = gemini_settings_auth(&gemini_home) {
104        return AuthResult { provider: "gemini".to_string(), status, auth_file: None, auth_age: None };
105    }
106    AuthResult { provider: "gemini".to_string(), status: AuthStatus::Installed, auth_file: None, auth_age: None }
107}

109pub fn check_antigravity(paths: &DuctorPaths) -> AuthResult {
110    if std::env::var("TUNER_TEST_MODE").is_ok() {
111        return AuthResult { provider: "antigravity".to_string(), status: AuthStatus::NotFound, auth_file: None, auth_age: None };
112    }
113    let binary_found = std::process::Command::new("agy").arg("models").output().is_ok();
114    if binary_found && check_antigravity_cli_logged_in() {
115        return AuthResult { provider: "antigravity".to_string(), status: AuthStatus::Authenticated, auth_file: None, auth_age: None };
116    }
117    let home = get_home_dir(paths);
118    let ccs_settings = home.join(".ccs").join("agy.settings.json");
119    let status = if binary_found || ccs_settings.is_file() { AuthStatus::Installed } else { AuthStatus::NotFound };
120    AuthResult { provider: "antigravity".to_string(), status, auth_file: None, auth_age: None }
121}

123fn get_home_dir(paths: &DuctorPaths) -> PathBuf {
124    if paths.tuner_home.ends_with(".tuner") {
125        paths.tuner_home.parent().unwrap_or(&paths.tuner_home).to_path_buf()
126    } else {
127        paths.tuner_home.clone()
128    }
129}

131fn check_claude_cli_logged_in() -> bool {
132    if std::env::var("TUNER_TEST_MODE").is_ok() {
133        return false;
134    }
135    match std::process::Command::new("claude").args(&["auth", "status"]).output() {
136        Ok(output) if output.status.success() => {
137            if let Ok(stdout_str) = std::str::from_utf8(&output.stdout) {
138                if let Ok(val) = serde_json::from_str::<serde_json::Value>(stdout_str) {
139                    return val.get("loggedIn") == Some(&serde_json::Value::Bool(true));
140                }
141            }
142            false
143        }
144        _ => false,
145    }
146}

148fn find_gemini_cli() -> bool {
149    if std::env::var("TUNER_TEST_MODE").is_ok() {
150        return false;
151    }
152    if std::process::Command::new("gemini").arg("--version").output().is_ok() {
153        return true;
154    }
155    if let Some(home_str) = std::env::var("HOME").ok() {
156        let home = Path::new(&home_str);
157        let versions_dir = home.join(".nvm").join("versions").join("node");
158        if versions_dir.is_dir() {
159            if let Ok(entries) = std::fs::read_dir(versions_dir) {
160                for entry in entries {
161                    if let Ok(entry) = entry {
162                        if entry.path().join("bin").join("gemini").is_file() {
163                            return true;
164                        }
165                    }
166                }
167            }
168        }
169    }
170    false
171}

173fn read_dotenv_has_gemini_keys(path: &Path) -> bool {
174    if let Ok(content) = std::fs::read_to_string(path) {
175        for line in content.lines() {
176            let trimmed = line.trim();
177            if trimmed.is_empty() || trimmed.starts_with('#') { continue; }
178            let clean = if trimmed.starts_with("export ") { trimmed["export ".len()..].trim() } else { trimmed };
179            if let Some(idx) = clean.find('=') {
180                let key = clean[..idx].trim();
181                let value = clean[idx + 1..].trim();
182                if (key == "GEMINI_API_KEY" || key == "GOOGLE_API_KEY") && !value.is_empty() && value != "\"\"" && value != "''" {
183                    return true;
184                }
185            }
186        }
187    }
188    false
189}

191fn config_has_gemini_key(paths: &DuctorPaths) -> bool {
192    let path = paths.config_path();
193    if let Ok(content) = std::fs::read_to_string(&path) {
194        if let Ok(val) = serde_json::from_str::<serde_json::Value>(&content) {
195            if let Some(key) = val.get("gemini_api_key").and_then(|v| v.as_str()) {
196                return !key.trim().is_empty();
197            }
198        }
199    }
200    false
201}

203fn gemini_settings_auth(gemini_home: &Path) -> Option<AuthStatus> {
204    let settings_file = gemini_home.join("settings.json");
205    if let Ok(content) = std::fs::read_to_string(&settings_file) {
206        if let Ok(val) = serde_json::from_str::<serde_json::Value>(&content) {
207            let selected_type = val.pointer("/security/auth/selectedType").and_then(|v| v.as_str());
208            if let Some(sel) = selected_type {
209                if sel == "oauth-personal" {
210                    let accounts_file = gemini_home.join("google_accounts.json");
211                    if let Ok(accs_content) = std::fs::read_to_string(&accounts_file) {
212                        if let Ok(accs_val) = serde_json::from_str::<serde_json::Value>(&accs_content) {
213                            if let Some(active) = accs_val.get("active").and_then(|v| v.as_str()) {
214                                if !active.trim().is_empty() {
215                                    return Some(AuthStatus::Authenticated);
216                                }
217                            }
218                        }
219                    }
220                } else if sel == "gemini-api-key" || sel == "vertex-ai" || sel == "compute-default-credentials" || sel == "cloud-shell" {
221                    return Some(AuthStatus::Authenticated);
222                }
223            }
224        }
225    }
226    None
227}

229fn check_antigravity_cli_logged_in() -> bool {
230    match std::process::Command::new("agy").arg("models").output() {
231        Ok(output) => {
232            let merged = format!("{}\n{}", 
233                std::str::from_utf8(&output.stdout).unwrap_or_default(),
234                std::str::from_utf8(&output.stderr).unwrap_or_default()
235            ).to_lowercase();
236            if merged.contains("sign in") || merged.contains("not logged in") || merged.contains("login") {
237                return false;
238            }
239            output.status.success()
240        }
241        _ => false,
242    }
