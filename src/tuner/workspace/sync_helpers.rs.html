12pub fn is_zone2_py_file(entry: &Path, src: &Path, root_src: &Path) -> bool {
13    if entry.extension().and_then(|s| s.to_str()) != Some("py") {
14        return false;
15    }
16    if let Ok(rel_dir) = src.strip_prefix(root_src) {
17        let rel_str = rel_dir.to_string_lossy().replace("\\", "/");
18        let zone2_py_dirs = [
19            "workspace/tools/cron_tools",
20            "workspace/tools/webhook_tools",
21            "workspace/tools/agent_tools",
22            "workspace/tools/task_tools",
23            "workspace/tools/media_tools",
24        ];
25        return zone2_py_dirs.contains(&rel_str.as_str());
26    }
27    false
28}

30pub fn walk_and_copy(src: &Path, dst: &Path, root_src: &Path) -> Result<(), String> {
31    if let Err(e) = std::fs::create_dir_all(dst) {
32        return Err(format!("Failed to create dir: {}", e));
33    }
34    let skip_dirs = [".venv", ".git", ".mypy_cache", "__pycache__", "node_modules"];
35    let skip_files = [
36        "RULES-claude-only.md",
37        "RULES-codex-only.md",
38        "RULES-gemini-only.md",
39        "RULES-all-clis.md",
40        "RULES.md",
41    ];
42
43    let entries = match std::fs::read_dir(src) {
44        Ok(e) => e,
45        Err(e) => return Err(format!("Failed to read dir: {}", e)),
46    };
47
48    for entry in entries {
49        if let Ok(entry) = entry {
50            let path = entry.path();
51            let name = match path.file_name().and_then(|s| s.to_str()) {
52                Some(n) => n,
53                None => continue,
54            };
55            if name.starts_with('.') || skip_dirs.contains(&name) || skip_files.contains(&name) {
56                continue;
57            }
58            let target = dst.join(name);
59            handle_entry(&path, &target, name, src, root_src, dst)?;
60        }
61    }
62    Ok(())
63}

65fn handle_entry(path: &Path, target: &Path, name: &str, src: &Path, root_src: &Path, dst: &Path) -> Result<(), String> {
66    if path.is_dir() {
67        if !target.is_symlink() {
68            walk_and_copy(path, target, root_src)?;
69        }
70    } else if name == "CLAUDE.md" || name == "AGENTS.md" || name == "GEMINI.md" {
71        if target.is_symlink() || target.is_file() {
72            let _ = std::fs::remove_file(target);
73        }
74        let _ = std::fs::copy(path, target);
75        if name == "CLAUDE.md" {
76            for mirror in &["AGENTS.md", "GEMINI.md"] {
77                let mirror_target = dst.join(mirror);
78                if mirror_target.is_symlink() || mirror_target.is_file() {
79                    let _ = std::fs::remove_file(&mirror_target);
80                }
81                let _ = std::fs::copy(path, &mirror_target);
82            }
83        }
84    } else if is_zone2_py_file(path, src, root_src) {
85        if target.is_file() {
86            let same_content = match (std::fs::read(path), std::fs::read(target)) {
87                (Ok(a), Ok(b)) => a == b,
88                _ => false,
89            };
90            if !same_content {
91                let backup = target.with_extension("py.bak");
92                let _ = std::fs::rename(target, &backup);
93            }
94        }
95        if target.is_symlink() {
96            let _ = std::fs::remove_file(target);
97        }
98        let _ = std::fs::copy(path, target);
99    } else if !target.exists() {
100        let _ = std::fs::copy(path, target);
101    }
102    Ok(())
103}

105pub fn sync_rule_files_recursive(dir: &Path) -> Result<(), String> {
106    let entries = match std::fs::read_dir(dir) {
107        Ok(e) => e,
108        Err(e) => return Err(format!("Failed to read dir: {}", e)),
109    };
110    let skip_dirs = [".venv", ".git", ".mypy_cache", "__pycache__", "node_modules"];
111
112    for entry in entries {
113        if let Ok(entry) = entry {
114            let path = entry.path();
115            if path.is_dir() {
116                if let Some(name) = path.file_name().and_then(|s| s.to_str()) {
117                    if skip_dirs.contains(&name) || name.starts_with('.') {
118                        continue;
119                    }
120                }
121                sync_group(&path)?;
122                sync_rule_files_recursive(&path)?;
123            }
124        }
125    }
126    Ok(())
127}

129pub fn sync_group(directory: &Path) -> Result<(), String> {
130    let filenames = ["CLAUDE.md", "AGENTS.md", "GEMINI.md"];
131    let mut existing = Vec::new();
132
133    for name in &filenames {
134        let path = directory.join(name);
135        if path.is_file() {
136            if let Ok(meta) = std::fs::metadata(&path) {
137                if let Ok(mtime) = meta.modified() {
138                    existing.push((*name, path, mtime));
139                }
140            }
141        }
142    }
143    if existing.is_empty() {
144        return Ok(());
145    }
146
147    let (newest_name, newest_path, newest_mtime) = existing
148        .iter()
149        .max_by_key(|(_, _, mtime)| mtime)
150        .unwrap();
151
152    for name in &filenames {
153        if *name == *newest_name { continue; }
154        let target_path = directory.join(name);
155        if target_path.is_file() {
156            let is_older = if let Ok(meta) = std::fs::metadata(&target_path) {
157                if let Ok(mtime) = meta.modified() {
158                    mtime < *newest_mtime
159                } else { true }
160            } else { true };
161
162            if is_older {
163                if let Err(e) = std::fs::copy(newest_path, &target_path) {
164                    return Err(format!("Failed to copy rule file: {}", e));
165                }
166                let ft_newest = filetime::FileTime::from_last_modification_time(&std::fs::metadata(newest_path).unwrap());
167                let _ = filetime::set_file_mtime(&target_path, ft_newest);
168            }
169        }
170    }
171    Ok(())
172}

174pub fn smart_merge_config(paths: &DuctorPaths) -> Result<(), String> {
175    let example_path = paths.config_example_path();
176    if !example_path.is_file() {
177        return Ok(());
178    }
179    let defaults_content = match std::fs::read_to_string(&example_path) {
180        Ok(c) => c,
181        Err(_) => return Ok(()),
182    };
183    let defaults: serde_json::Value = match serde_json::from_str(&defaults_content) {
184        Ok(v) => v,
185        Err(_) => return Ok(()),
186    };
187
188    let config_path = paths.config_path();
189    if !config_path.is_file() {
190        if let Some(parent) = config_path.parent() {
191            let _ = std::fs::create_dir_all(parent);
192        }
193        if let Ok(pretty) = serde_json::to_string_pretty(&defaults) {
194            let _ = std::fs::write(&config_path, pretty);
195        }
196    }
197    Ok(())
198}

200pub fn smart_merge_profile_config(paths: &DuctorPaths) -> Result<(), String> {
201    let example_path = paths.home_defaults.join("config").join("config.json");
202    if !example_path.is_file() {
203        return Ok(());
204    }
205    let defaults_content = std::fs::read_to_string(&example_path).map_err(|e| e.to_string())?;
206    let defaults: serde_json::Value = serde_json::from_str(&defaults_content).map_err(|e| e.to_string())?;
207
208    let config_path = paths.profile_home().join("config").join("config.json");
209    if !config_path.is_file() {
210        if let Some(parent) = config_path.parent() {
211            let _ = std::fs::create_dir_all(parent);
212        }
213        if let Ok(pretty) = serde_json::to_string_pretty(&defaults) {
214            let _ = std::fs::write(&config_path, pretty);
215        }
216    }
217    Ok(())
