1//! # Workspace Rule Profile Selector

3//! ## Overview

6//! ## Collaboration Graph
7//! - Called by [`init_workspace`](super::sync::init_workspace) to sync rule configurations.

12use crate::workspace::paths::DuctorPaths;
13use crate::workspace::rules_check::{check_antigravity, check_claude, check_codex, check_gemini};
14use std::path::{Path, PathBuf};

16/// Authentication status for CLI providers.
17#[derive(Clone, Copy, Debug, PartialEq, Eq)]
18pub enum AuthStatus {
19    Authenticated,
20    Installed,
21    NotFound,
22}
23
24/// Verification result for a provider.
25#[derive(Clone, Debug)]
26pub struct AuthResult {
27    pub provider: String,
28    pub status: AuthStatus,
29    pub auth_file: Option<PathBuf>,
30    pub auth_age: Option<chrono::DateTime<chrono::Utc>>,
31}
32
33/// Selector and deployer for variant-specific rule files.
34pub struct RulesSelector {
35    pub paths: DuctorPaths,
36    pub claude_authenticated: bool,
37    pub codex_authenticated: bool,
38    pub gemini_authenticated: bool,
39    pub antigravity_authenticated: bool,
40}
41
42impl RulesSelector {
43    /// Discovers auth status and returns a RulesSelector instance.
44    pub fn new(paths: DuctorPaths) -> Self {
45        let claude = check_claude(&paths);
46        let codex = check_codex(&paths);
47        let gemini = check_gemini(&paths);
48        let agy = check_antigravity(&paths);
49        Self {
50            paths,
51            claude_authenticated: claude.status == AuthStatus::Authenticated,
52            codex_authenticated: codex.status == AuthStatus::Authenticated,
53            gemini_authenticated: gemini.status == AuthStatus::Authenticated,
54            antigravity_authenticated: agy.status == AuthStatus::Authenticated,
55        }
56    }
57
58    /// Determines variant suffix (e.g. "claude-only", "all-clis", "gemini-only").
59    pub fn get_variant_suffix(&self) -> String {
60        let auth_count = self.claude_authenticated as usize
61            + self.codex_authenticated as usize
62            + self.gemini_authenticated as usize
63            + self.antigravity_authenticated as usize;
64
65        if auth_count >= 2 {
66            "all-clis".to_string()
67        } else if self.codex_authenticated {
68            "codex-only".to_string()
69        } else if self.antigravity_authenticated {
70            "antigravity".to_string()
71        } else if self.gemini_authenticated {
72            "gemini-only".to_string()
73        } else {
74            "claude-only".to_string()
75        }
76    }
77
78    /// Discovers directories in `home_defaults` containing templates.
79    pub fn discover_template_directories(&self) -> Vec<PathBuf> {
80        let mut candidates = Vec::new();
81        find_rules_templates(&self.paths.home_defaults, &mut candidates);
82        candidates.sort();
83        candidates
84    }
85
86    /// Selects the best template (variant or static) for a directory.
87    pub fn get_best_template(&self, directory: &Path) -> Option<PathBuf> {
88        let variant = self.get_variant_suffix();
89        let variant_template = directory.join(format!("RULES-{}.md", variant));
90        if variant_template.is_file() {
91            return Some(variant_template);
92        }
93        if variant == "antigravity" {
94            let gemini_template = directory.join("RULES-gemini-only.md");
95            if gemini_template.is_file() {
96                return Some(gemini_template);
97            }
98        }
99        let static_template = directory.join("RULES.md");
100        if static_template.is_file() {
101            return Some(static_template);
102        }
103        None
104    }
105
106    /// Deploys rule files to matching locations.
107    pub fn deploy_rules(&self) -> Result<(), String> {
108        let template_dirs = self.discover_template_directories();
109        for template_dir in template_dirs {
110            let rel_path = match template_dir.strip_prefix(&self.paths.home_defaults) {
111                Ok(p) => p,
112                Err(_) => continue,
113            };
114            let template = match self.get_best_template(&template_dir) {
115                Some(t) => t,
116                None => continue,
117            };
118            let dst_dir = self.paths.profile_home().join(rel_path);
119            if let Err(e) = std::fs::create_dir_all(&dst_dir) {
120                return Err(format!("Failed to create directory {:?}: {}", dst_dir, e));
121            }
122            if self.claude_authenticated {
123                let _ = std::fs::copy(&template, dst_dir.join("CLAUDE.md"));
124            }
125            if self.codex_authenticated || self.antigravity_authenticated {
126                let _ = std::fs::copy(&template, dst_dir.join("AGENTS.md"));
127            }
128            if self.gemini_authenticated || self.antigravity_authenticated {
129                let _ = std::fs::copy(&template, dst_dir.join("GEMINI.md"));
130            }
131        }
132        let _ = self.cleanup_stale_files();
133        Ok(())
134    }
135
136    /// Removes rule files for unauthenticated CLI providers.
137    pub fn cleanup_stale_files(&self) -> Result<usize, String> {
138        let mut stale = Vec::new();
139        if !self.claude_authenticated {
140            stale.push("CLAUDE.md");
141        }
142        if !self.codex_authenticated && !self.antigravity_authenticated {
143            stale.push("AGENTS.md");
144        }
145        if !self.gemini_authenticated && !self.antigravity_authenticated {
146            stale.push("GEMINI.md");
147        }
148        let mut total_removed = 0;
149        for filename in stale {
150            total_removed += self.remove_files_by_name(filename)?;
151        }
152        Ok(total_removed)
153    }
154
155    /// Removes files with the given name recursively, protecting cron task directories.
156    pub fn remove_files_by_name(&self, filename: &str) -> Result<usize, String> {
157        let cron_tasks_path = self.paths.workspace().join("cron_tasks");
158        let mut count = 0;
159        remove_files_recursively(&self.paths.profile_home(), filename, &cron_tasks_path, &mut count);
160        Ok(count)
161    }
162}
163
164fn find_rules_templates(dir: &Path, candidates: &mut Vec<PathBuf>) {
165    if !dir.is_dir() { return; }
166    let mut has_rules = false;
167    if let Ok(entries) = std::fs::read_dir(dir) {
168        for entry in entries {
169            if let Ok(entry) = entry {
170                let path = entry.path();
171                if path.is_dir() {
172                    find_rules_templates(&path, candidates);
173                } else if let Some(name) = path.file_name().and_then(|s| s.to_str()) {
174                    if name.starts_with("RULES") && name.ends_with(".md") {
175                        has_rules = true;
176                    }
177                }
178            }
179        }
180    }
181    if has_rules {
182        candidates.push(dir.to_path_buf());
183    }
184}
185
186fn remove_files_recursively(dir: &Path, filename: &str, cron_tasks_path: &Path, count: &mut usize) {
187    if !dir.is_dir() { return; }
188    if let Ok(entries) = std::fs::read_dir(dir) {
189        for entry in entries {
190            if let Ok(entry) = entry {
191                let path = entry.path();
192                if path.is_dir() {
193                    if path.starts_with(cron_tasks_path) { continue; }
194                    remove_files_recursively(&path, filename, cron_tasks_path, count);
195                } else if let Some(name) = path.file_name().and_then(|s| s.to_str()) {
196                    if name == filename {
197                        if !path.starts_with(cron_tasks_path) {
198                            if std::fs::remove_file(&path).is_ok() {
199                                *count += 1;
200                            }
201                        }
202                    }
203                }
204            }
205        }
206    }
