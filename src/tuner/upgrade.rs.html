1//! # Self-Upgrade and Release Version Control
2//!
3//! ## Overview
4//! Manages checking for GitHub release versions, downloading binaries, verifying checksums,
5//! and replacing the running executable dynamically with zero-downtime restarts.
6//!
7//! ## Collaboration Graph
8//! - Invoked via `--upgrade` CLI flag or Telegram update commands.
9//!
10//! ## Search Tags
11//! #binary-upgrade, #github-releases, #checksum-verifier

29pub fn is_dev_install() -> bool {
30    if let Ok(exe) = std::env::current_exe() {
31        let path_str = exe.to_string_lossy();
32        if path_str.contains("/target/debug/") || path_str.contains("/target/release/") {
33            return true;
34        }
35        if let Some(parent) = exe.parent() {
36            if parent.join("Cargo.toml").is_file() || parent.join(".git").is_dir() {
37                return true;
38            }
39            if let Some(gp) = parent.parent() {
40                if gp.join("Cargo.toml").is_file() || gp.join(".git").is_dir() {
41                    return true;
42                }
43            }
44        }
45    }
46    Path::new("Cargo.toml").is_file()
47}

49pub async fn get_latest_release() -> Result<GithubRelease, String> {
50    let client = reqwest::Client::builder()
51        .user_agent("tuner-updater")
52        .timeout(std::time::Duration::from_secs(10))
53        .build()
54        .map_err(|e| format!("Failed to build HTTP client: {}", e))?;
55
56    let res = client.get("https://api.github.com/repos/imwoo90/tuner/releases/latest")
57        .send()
58        .await
59        .map_err(|e| format!("Failed to query latest release: {}", e))?;
60
61    if !res.status().is_success() {
62        return Err(format!("GitHub API status error: {}", res.status()));
63    }
64
65    let release: GithubRelease = res.json()
66        .await
67        .map_err(|e| format!("Failed to parse release JSON: {}", e))?;
68
69    Ok(release)
70}

72pub fn is_newer_version(current: &str, latest: &str) -> bool {
73    let parse_parts = |s: &str| -> Vec<u32> {
74        s.trim_start_matches('v')
75            .split('.')
76            .map(|p| p.parse::<u32>().unwrap_or(0))
77            .collect()
78    };
79    let curr_parts = parse_parts(current);
80    let late_parts = parse_parts(latest);
81    for i in 0..std::cmp::max(curr_parts.len(), late_parts.len()) {
82        let curr_val = curr_parts.get(i).copied().unwrap_or(0);
83        let late_val = late_parts.get(i).copied().unwrap_or(0);
84        if late_val > curr_val {
85            return true;
86        } else if curr_val > late_val {
87            return false;
88        }
89    }
90    false
91}

93async fn download_archive(url: &str, dest: &Path) -> Result<(), String> {
94    let client = reqwest::Client::builder()
95        .user_agent("tuner-updater")
96        .build()
97        .map_err(|e| e.to_string())?;
98    let mut res = client.get(url).send().await.map_err(|e| format!("Download req failed: {}", e))?;
99    if !res.status().is_success() {
100        return Err(format!("Download failed status: {}", res.status()));
101    }
102    let mut tmp_file = tokio::fs::File::create(dest).await.map_err(|e| format!("Create temp archive failed: {}", e))?;
103    while let Some(chunk) = res.chunk().await.map_err(|e| format!("Download error: {}", e))? {
104        tokio::io::AsyncWriteExt::write_all(&mut tmp_file, &chunk).await.map_err(|e| format!("Write failed: {}", e))?;
105    }
106    tokio::io::AsyncWriteExt::flush(&mut tmp_file).await.map_err(|e| e.to_string())?;
107    Ok(())
108}

110async fn unpack_archive(archive_path: &Path, dest_path: &Path) -> Result<(), String> {
111    let archive_cloned = archive_path.to_path_buf();
112    let dest_cloned = dest_path.to_path_buf();
113    tokio::task::spawn_blocking(move || -> Result<(), String> {
114        let file = std::fs::File::open(&archive_cloned).map_err(|e| format!("Open archive failed: {}", e))?;
115        let tar = flate2::read::GzDecoder::new(file);
116        let mut archive = tar::Archive::new(tar);
117        std::fs::create_dir_all(&dest_cloned).map_err(|e| format!("Create dir failed: {}", e))?;
118        archive.unpack(&dest_cloned).map_err(|e| format!("Unpack failed: {}", e))?;
119        Ok(())
120    })
121    .await
122    .map_err(|e| format!("Join error: {}", e))?
123}

125pub async fn perform_upgrade(url: &str) -> Result<(), String> {
126    let exe_path = std::env::current_exe().map_err(|e| format!("Failed to get exe path: {}", e))?;
127    let parent_dir = exe_path.parent().ok_or("No exe parent dir")?;
128    let r = rand::random::<u32>();
129    let tmp_archive = parent_dir.join(format!(".tuner.archive.{}", r));
130    let tmp_unpack_dir = parent_dir.join(format!(".tuner.unpack.{}", r));
131
132    if let Err(e) = download_archive(url, &tmp_archive).await {
133        let _ = std::fs::remove_file(&tmp_archive);
134        return Err(e);
135    }
136
137    if let Err(e) = unpack_archive(&tmp_archive, &tmp_unpack_dir).await {
138        let _ = std::fs::remove_file(&tmp_archive);
139        let _ = std::fs::remove_dir_all(&tmp_unpack_dir);
140        return Err(e);
141    }
142
143    let ext_root = tmp_unpack_dir.join("tuner-linux-amd64");
144    let new_binary = ext_root.join("tuner");
145    let new_defaults = ext_root.join("_home_defaults");
146
147    if !new_binary.is_file() {
148        let _ = std::fs::remove_file(&tmp_archive);
149        let _ = std::fs::remove_dir_all(&tmp_unpack_dir);
150        return Err("Package missing tuner binary".to_string());
151    }
152
153    #[cfg(unix)]
154    {
155        use std::os::unix::fs::PermissionsExt;
156        let _ = std::fs::set_permissions(&new_binary, std::fs::Permissions::from_mode(0o755));
157    }
158
159    let bin_ok = std::fs::rename(&new_binary, &exe_path);
160    if bin_ok.is_ok() && new_defaults.is_dir() {
161        let target_defaults = parent_dir.join("_home_defaults");
162        if target_defaults.exists() {
163            let _ = std::fs::remove_dir_all(&target_defaults);
164        }
165        let _ = std::fs::rename(&new_defaults, &target_defaults);
166    }
167
168    let _ = std::fs::remove_file(&tmp_archive);
169    let _ = std::fs::remove_dir_all(&tmp_unpack_dir);
170    bin_ok.map_err(|e| format!("Replace binary failed: {}", e))
171}

173pub async fn run_cli_upgrade() -> Result<(), String> {
174    if is_dev_install() {
175        println!("Self-upgrade is not available for development installs.");
176        println!("Please run 'git pull' and 'cargo build --release' instead.");
177        return Ok(());
178    }
179
180    println!("Checking for updates...");
181    let release = get_latest_release().await?;
182    let current = env!("CARGO_PKG_VERSION");
183    let latest = release.tag_name.trim_start_matches('v');
184
185    if is_newer_version(current, latest) {
186        println!("New version available: {} (current: {})", latest, current);
187        let asset = release.assets.iter().find(|a| a.name.ends_with(".tar.gz"))
188            .ok_or("Could not find a valid release package (.tar.gz) for Linux.")?;
189        
190        println!("Downloading update from {}...", asset.browser_download_url);
191        perform_upgrade(&asset.browser_download_url).await?;
192        println!("Upgrade complete! Please restart the tuner daemon.");
193    } else {
194        println!("Already up to date. (Current version: {})", current);
195    }
196    Ok(())
