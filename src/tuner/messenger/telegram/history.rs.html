1//! # Telegram Chat History logger
2//!
3//! ## Overview
4//! Manages persistent logging of message histories (role, sender, content, thread details)
5//! into jsonl files for trace audits, session resume context, and debugging.
6//!
7//! ## Collaboration Graph
8//! - Log entries saved under workspace directories.
9//! - Consulted during session initialization to restore history context.
10//!
11//! ## Search Tags
12//! #history-logger, #trace-auditing, #jsonl-logs
13
14use std::path::Path;
15use std::fs::{OpenOptions, create_dir_all};
16use std::io::Write;
17use chrono::Utc;
18use serde::Serialize;
19
20#[derive(Serialize)]
21pub struct TelegramHistoryEntry {
22    pub timestamp: String,
23    pub sender: String,
24    pub message_id: Option<i32>,
25    pub text: String,
26    pub is_success: bool,
27    pub error: Option<String>,
28}
29
30pub fn log_telegram_message(
31    working_dir: &Path,
32    session_id: &str,
33    sender: &str,
34    message_id: Option<i32>,
35    text: &str,
36    is_success: bool,
37    error: Option<&str>,
38) {
39    if session_id.is_empty() {
40        return;
41    }
42    let target_dir = working_dir.join("brain").join(session_id);
43    if let Err(e) = create_dir_all(&target_dir) {
44        eprintln!("Failed to create directory for telegram history: {:?}", e);
45        return;
46    }
47    let file_path = target_dir.join("telegram_history.jsonl");
48    let entry = TelegramHistoryEntry {
49        timestamp: Utc::now().to_rfc3339(),
50        sender: sender.to_string(),
51        message_id,
52        text: text.to_string(),
53        is_success,
54        error: error.map(|s| s.to_string()),
55    };
56    if let Ok(serialized) = serde_json::to_string(&entry) {
57        if let Ok(mut file) = OpenOptions::new()
58            .create(true)
59            .append(true)
60            .open(&file_path)
61        {
62            let _ = writeln!(file, "{}", serialized);
63        }
64    }
65}
