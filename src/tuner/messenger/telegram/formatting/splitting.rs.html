12fn split_at_char_limit(s: &str, char_limit: usize) -> (&str, &str) {
13    if s.is_empty() || char_limit == 0 {
14        return ("", s);
15    }
16    let mut char_count = 0;
17    let mut byte_idx = 0;
18    for c in s.chars() {
19        if char_count >= char_limit {
20            break;
21        }
22        char_count += 1;
23        byte_idx += c.len_utf8();
24    }
25    s.split_at(byte_idx)
26}

28fn get_tag_name(tag: &str) -> &str {
29    let stripped = tag.trim_matches(|c| c == '<' || c == '>');
30    stripped.split_whitespace().next().unwrap_or("").trim_start_matches('/')
31}

33fn get_closing_tags(open_tags: &[String]) -> String {
34    open_tags.iter().rev()
35        .map(|t| format!("</{}>", get_tag_name(t)))
36        .collect::<Vec<_>>()
37        .join("")
38}

40fn tokenize_html(text: &str) -> Vec<&str> {
41    let tok_re = Regex::new(r"(</?[a-zA-Z][^>]*>|\n\n|\n)").unwrap();
42    let mut tokens = Vec::new();
43    let mut last_idx = 0;
44
45    for mat in tok_re.find_iter(text) {
46        if mat.start() > last_idx {
47            tokens.push(&text[last_idx..mat.start()]);
48        }
49        tokens.push(mat.as_str());
50        last_idx = mat.end();
51    }
52    if last_idx < text.len() {
53        tokens.push(&text[last_idx..]);
54    }
55    tokens
56}

58fn split_oversized_token(
59    mut token: &str,
60    max_len: usize,
61    open_tags: &[String],
62    current_chunk_parts: &mut Vec<String>,
63    current_len: &mut usize,
64    chunks: &mut Vec<String>,
65) -> String {
66    let char_len = |s: &str| s.chars().count();
67    let mut closing_tags_str = get_closing_tags(open_tags);
68    let mut closing_tags_len = char_len(&closing_tags_str);
69
70    while char_len(token) > (max_len.saturating_sub(*current_len).saturating_sub(closing_tags_len)) {
71        let available = max_len.saturating_sub(*current_len).saturating_sub(closing_tags_len);
72        let take = if available == 0 { 1 } else { available };
73        let (prefix, suffix) = split_at_char_limit(token, take);
74        current_chunk_parts.push(prefix.to_string());
75        token = suffix;
76
77        current_chunk_parts.push(closing_tags_str.clone());
78        chunks.push(current_chunk_parts.join(""));
79
80        *current_chunk_parts = open_tags.to_vec();
81        *current_len = open_tags.iter().map(|t| char_len(t)).sum();
82        closing_tags_str = get_closing_tags(open_tags);
83        closing_tags_len = char_len(&closing_tags_str);
84    }
85    token.to_string()
86}

88fn update_tags_and_len(
89    token: &str,
90    is_tag: bool,
91    open_tags: &mut Vec<String>,
92    current_chunk_parts: &mut Vec<String>,
93    current_len: &mut usize,
94) {
95    if token.is_empty() {
96        return;
97    }
98    current_chunk_parts.push(token.to_string());
99    *current_len += token.chars().count();
100
101    if is_tag {
102        if token.starts_with("</") {
103            let name = get_tag_name(token);
104            if let Some(pos) = open_tags.iter().rposition(|t| get_tag_name(t) == name) {
105                open_tags.remove(pos);
106            }
107        } else if !token.ends_with("/>") {
108            open_tags.push(token.to_string());
109        }
110    }
111}

113fn process_token(
114    token_ref: &str,
115    max_len: usize,
116    open_tags: &mut Vec<String>,
117    current_chunk_parts: &mut Vec<String>,
118    current_len: &mut usize,
119    chunks: &mut Vec<String>,
120) {
121    let char_len = |s: &str| s.chars().count();
122    let mut token = token_ref.to_string();
123    if token.is_empty() {
124        return;
125    }
126
127    let is_tag = token.starts_with('<') && token.ends_with('>');
128    let closing_tags_str = get_closing_tags(open_tags);
129    let closing_tags_len = char_len(&closing_tags_str);
130
131    if !is_tag && char_len(&token) > (max_len.saturating_sub(*current_len).saturating_sub(closing_tags_len)) {
132        let available = max_len.saturating_sub(*current_len).saturating_sub(closing_tags_len);
133        if available > 0 {
134            let (prefix, suffix) = split_at_char_limit(&token, available);
135            current_chunk_parts.push(prefix.to_string());
136            token = suffix.to_string();
137        }
138
139        current_chunk_parts.push(closing_tags_str.clone());
140        chunks.push(current_chunk_parts.join(""));
141
142        *current_chunk_parts = open_tags.clone();
143        *current_len = open_tags.iter().map(|t| char_len(t)).sum();
144
145        token = split_oversized_token(&token, max_len, open_tags, current_chunk_parts, current_len, chunks);
146    } else if !current_chunk_parts.is_empty() && (*current_len + char_len(&token) + closing_tags_len > max_len) {
147        current_chunk_parts.push(closing_tags_str);
148        chunks.push(current_chunk_parts.join(""));
149
150        *current_chunk_parts = open_tags.clone();
151        *current_len = open_tags.iter().map(|t| char_len(t)).sum();
152    }
153
154    update_tags_and_len(&token, is_tag, open_tags, current_chunk_parts, current_len);
155}

157fn has_text_content(s: &str) -> bool {
158    let re = Regex::new(r"<[^>]*>").unwrap();
159    let text = re.replace_all(s, "");
160    !text.trim().is_empty()
161}

163pub fn split_html_message(text: &str, max_len: usize) -> Vec<String> {
164    let char_len = |s: &str| s.chars().count();
165    if char_len(text) <= max_len {
166        if text.is_empty() {
167            return vec![];
168        }
169        return vec![text.to_string()];
170    }
171
172    let tokens = tokenize_html(text);
173    let mut chunks = Vec::new();
174    let mut current_chunk_parts: Vec<String> = Vec::new();
175    let mut current_len = 0;
176    let mut open_tags: Vec<String> = Vec::new();
177
178    for token_ref in tokens {
179        process_token(
180            token_ref,
181            max_len,
182            &mut open_tags,
183            &mut current_chunk_parts,
184            &mut current_len,
185            &mut chunks,
186        );
187    }
188
189    if !current_chunk_parts.is_empty() {
190        let closing_tags_str = get_closing_tags(&open_tags);
191        current_chunk_parts.push(closing_tags_str);
192        chunks.push(current_chunk_parts.join(""));
193    }
194
195    chunks.into_iter().filter(|c| has_text_content(c)).collect()
196}
