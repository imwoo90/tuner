19pub fn get_topic_id(msg: &Message) -> Option<i64> {
20    match &msg.kind {
21        teloxide::types::MessageKind::Common(c) if c.is_topic_message => msg.thread_id.map(|t| t as i64),
22        _ => None,
23    }

26pub(crate) fn strip_mention(text: &str, bot_username: Option<&str>) -> String {
27    let u = match bot_username {
28        Some(n) if !n.is_empty() => n,
29        _ => return text.to_string(),
30    };
31    let tag = format!("@{}", u.strip_prefix('@').unwrap_or(u));
32    let lower_text = text.to_lowercase();
33    let lower_tag = tag.to_lowercase();
34
35    if let Some(idx) = lower_text.find(&lower_tag) {
36        let before = &text[..idx];
37        let after = &text[idx + tag.len()..];
38        let trimmed = format!("{}{}", before, after).trim().to_string();
39        if trimmed.is_empty() { text.to_string() } else { trimmed }
40    } else {
41        text.to_string()
42    }
43}

47pub(crate) fn build_reply_prompt(message: &Message, user_text: &str) -> String {
48    let cited = if let Some(replied) = message.reply_to_message() {
49        replied.text().or(replied.caption()).map(|s| s.trim())
50    } else {
51        None
52    };
53
54    match cited {
55        None => user_text.to_string(),
56        Some("") => user_text.to_string(),
57        Some(text) => {
58            let quoted = text
59                .lines()
60                .map(|line| format!("> {}", line))
61                .collect::<Vec<String>>()
62                .join("\n");
63            format!(
64                "The user is replying to this quoted message:\n{}\n\nThe user's message:\n{}",
65                quoted, user_text
66            )
67        }
68    }
69}

73fn get_remaining_prompt(rest: &str, consumed_tokens: usize) -> &str {
74    let mut remaining_text = "";
75    let mut count = 0;
76    for (i, ch) in rest.char_indices() {
77        if ch.is_whitespace() {
78            if i > 0 && !rest.as_bytes()[i-1].is_ascii_whitespace() {
79                count += 1;
80                if count == consumed_tokens {
81                    remaining_text = rest[i..].trim();
82                    break;
83                }
84            }
85        }
86    }
87    if count < consumed_tokens {
88        ""
89    } else {
90        remaining_text
91    }
92}

94pub(crate) fn parse_model_directive(text: &str) -> (Option<String>, Option<String>, &str) {
95    let t = text.trim();
96    if let Some(r) = t.strip_prefix("@model ") {
97        let rest = r.trim();
98        let parts: Vec<&str> = rest.split_whitespace().collect();
99        if !parts.is_empty() {
100            let mut model_name = parts[0].to_string();
101            let mut effort = None;
102            let mut consumed_tokens = 1;
103            
104            if parts.len() >= 3 && parts[1] == "--effort" {
105                effort = Some(parts[2].to_string());
106                consumed_tokens = 3;
107            } else if parts.len() >= 2 && (parts[1] == "high" || parts[1] == "medium" || parts[1] == "low") {
108                effort = Some(parts[1].to_string());
109                consumed_tokens = 2;
110            }
111
112            if model_name.ends_with("-high") {
113                effort = Some("high".to_string());
114                model_name = model_name[..model_name.len() - 5].to_string();
115            } else if model_name.ends_with("-medium") {
116                effort = Some("medium".to_string());
117                model_name = model_name[..model_name.len() - 7].to_string();
118            } else if model_name.ends_with("-low") {
119                effort = Some("low".to_string());
120                model_name = model_name[..model_name.len() - 4].to_string();
121            }
122            
123            let remaining_text = get_remaining_prompt(rest, consumed_tokens);
124            return (Some(model_name), effort, remaining_text);
125        }
126    } else if t.starts_with('@') {
127        let dir = t.split_whitespace().next().unwrap_or("");
128        let m = &dir[1..];
129        if ["opus", "sonnet", "haiku", "gpt-4o", "gpt-4-turbo"].contains(&m) {
130            return (Some(m.to_string()), None, t[dir.len()..].trim());
131        }
132    }
133    (None, None, t)
134}

137pub(crate) async fn load_sessions_cache(
138    sessions: &crate::session::manager::SessionManager,
139    cache: &crate::telegram::TopicNameCache,
140) {
141    if let Ok(all) = sessions.list_all().await {
142        for s in all {
143            if let (Some(tid), Some(tname)) = (s.topic_id, s.topic_name) {
144                cache.insert(s.chat_id, tid, tname);
145            }
146        }
147    }
148}

150pub(crate) async fn resolve_session_model(
151    msg: &Message,
152    config: &crate::config::CliConfig,
153    sessions: &crate::session::manager::SessionManager,
154) -> String {
155    let topic_id = crate::telegram::get_topic_id(msg);
156    let key = crate::session::key::SessionKey::telegram(msg.chat.id.0, topic_id);
157    let default_model = config.model.as_deref().unwrap_or("antigravity-default");
158    if let Ok((sess, _)) = sessions.resolve_session(&key, &config.provider, default_model).await {
159        let eff = sess.effort.as_ref().or(config.effort.as_ref());
160        if let Some(eff_str) = eff {
161            if !eff_str.is_empty() {
162                return format!("{} (effort: {})", sess.model, eff_str);
163            }
164        }
165        sess.model
166    } else {
167        default_model.to_string()
168    }
169}

177pub(crate) async fn send_startup_notification(
178    bot: teloxide::Bot,
179    sessions: std::sync::Arc<crate::session::manager::SessionManager>,
180) {
181    println!("🤖 [tuner] send_startup_notification task started");
182    match sessions.list_all().await {
183        Ok(sess_list) => {
184            println!("🤖 [tuner] Loaded {} sessions for startup notification", sess_list.len());
185            if let Some(sess) = find_last_active_session(&sess_list) {
186                let active_lang = sess.language.clone().unwrap_or_else(|| "en".to_string());
187                crate::i18n::TASK_ACTIVE_LANG.scope(active_lang, async move {
188                    let startup_msg = crate::t!("bot.startup_complete");
189                    let mut req = bot.send_message(teloxide::types::ChatId(sess.chat_id), startup_msg);
190                    if let Some(tid) = sess.topic_id {
191                        req = req.message_thread_id(tid as i32);
192                    }
193                    match req.await {
194                        Ok(_) => println!("🤖 [tuner] Startup notification sent successfully to chat_id: {}", sess.chat_id),
195                        Err(e) => eprintln!("❌ [tuner] Failed to send startup notification to chat_id {}: {:?}", sess.chat_id, e),
196                    }
197                }).await;
198            } else {
199                println!("🤖 [tuner] No active Telegram session found for startup notification");
200            }
201        }
202        Err(e) => {
203            eprintln!("❌ [tuner] Failed to list sessions for startup notification: {:?}", e);
204        }
205    }
206}
