1//! # Telegram Typing Status Indicator Guard
2//!
3//! ## Overview
4//! Periodically sends the "typing..." chat action to Telegram while an agent is generating code
5//! or executing background tasks. Dropped automatically when execution ends.
6//!
7//! ## Collaboration Graph
8//! - Spawned during PTY execution streams.
9//! - Binds directly to the Teloxide client handle.
10//!
11//! ## Search Tags
12//! #typing-indicator, #status-guard, #user-feedback
13
14use std::time::Duration;
15use tokio::task::JoinHandle;
16use teloxide::prelude::*;
17use teloxide::types::ChatAction;
18
19/// RAII guard for Telegram typing activity and message reaction.
20pub struct TelegramTypingGuard {
21    token: String,
22    chat_id: ChatId,
23    message_id: teloxide::types::MessageId,
24    handle: JoinHandle<()>,
25}
26
27impl TelegramTypingGuard {
28    /// Create a new guard, setting the typing action and reactions.
29    pub async fn new(
30        bot: Bot,
31        token: String,
32        msg: &Message,
33    ) -> Self {
34        let chat_id = msg.chat.id;
35        let message_id = msg.id;
36        let thread_id = msg.thread_id;
37
38        let bot_clone = bot.clone();
39        tokio::spawn(async move {
40            let mut req = bot_clone.send_chat_action(chat_id, ChatAction::Typing);
41            if let Some(t) = thread_id {
42                req = req.message_thread_id(t);
43            }
44            let _ = req.await;
45        });
46
47        let token_clone = token.clone();
48        tokio::spawn(async move {
49            let client = reqwest::Client::new();
50            let url = format!("https://api.telegram.org/bot{}/setMessageReaction", token_clone);
51            let body = serde_json::json!({
52                "chat_id": chat_id.0,
53                "message_id": message_id.0,
54                "reaction": [
55                    {
56                        "type": "emoji",
57                        "emoji": "👀"
58                    }
59                ]
60            });
61            let _ = client.post(&url).json(&body).send().await;
62        });
63
64        let bot_clone3 = bot.clone();
65        let handle = tokio::spawn(async move {
66            loop {
67                tokio::time::sleep(Duration::from_secs(4)).await;
68                let mut req = bot_clone3.send_chat_action(chat_id, ChatAction::Typing);
69                if let Some(t) = thread_id {
70                    req = req.message_thread_id(t);
71                }
72                let _ = req.await;
73            }
74        });
75
76        Self {
77            token,
78            chat_id,
79            message_id,
80            handle,
81        }
82    }
83}
84
85impl Drop for TelegramTypingGuard {
86    fn drop(&mut self) {
87        self.handle.abort();
88        let token = self.token.clone();
89        let chat_id = self.chat_id;
90        let message_id = self.message_id;
91        tokio::spawn(async move {
92            let client = reqwest::Client::new();
93            let url = format!("https://api.telegram.org/bot{}/setMessageReaction", token);
94            let body = serde_json::json!({
95                "chat_id": chat_id.0,
96                "message_id": message_id.0,
97                "reaction": []
98            });
99            let _ = client.post(&url).json(&body).send().await;
100        });
101    }
102}
