16pub(crate) fn build_single_select_keyboard(
17    sess_id: &str,
18    options: &[String],
19    show_prev: bool,
20) -> teloxide::types::InlineKeyboardMarkup {
21    let mut keyboard = Vec::new();
22    for (i, opt) in options.iter().enumerate() {
23        let lower = opt.to_lowercase();
24        if lower.contains("write-in") || lower.contains("직접 입력") {
25            continue;
26        }
27        let callback_data = format!("ask_ans:{}:{}", sess_id, i);
28        keyboard.push(vec![teloxide::types::InlineKeyboardButton::callback(opt, callback_data)]);
29    }
30    if show_prev {
31        keyboard.push(vec![
32            teloxide::types::InlineKeyboardButton::callback("⬅️ 이전 질문 (Prev)", format!("ask_prev:{}", sess_id)),
33            teloxide::types::InlineKeyboardButton::callback("⏭️ 건너뛰기 (Skip)", format!("ask_skip:{}", sess_id)),
34        ]);
35    } else {
36        keyboard.push(vec![
37            teloxide::types::InlineKeyboardButton::callback("⏭️ 건너뛰기 (Skip)", format!("ask_skip:{}", sess_id)),
38        ]);
39    }
40    teloxide::types::InlineKeyboardMarkup::new(keyboard)
41}

43pub(crate) fn get_bitmap_for_question(
44    q: &crate::cli::AskQuestionData,
45    answers: &[String],
46    index: usize,
47) -> String {
48    if !q.is_multi_select {
49        return String::new();
50    }
51    if let Some(ans) = answers.get(index) {
52        if ans.is_empty() {
53            return "0".repeat(q.options.len());
54        }
55        let parts: Vec<&str> = ans.split(", ").map(|s| s.trim()).collect();
56        let mut bm = String::new();
57        for opt in &q.options {
58            if parts.contains(&opt.as_str()) {
59                bm.push('1');
60            } else {
61                bm.push('0');
62            }
63        }
64        bm
65    } else {
66        "0".repeat(q.options.len())
67    }
68}

70pub(crate) fn build_multi_select_keyboard(
71    sess_id: &str,
72    options: &[String],
73    bitmap: &str,
74    show_prev: bool,
75) -> teloxide::types::InlineKeyboardMarkup {
76    let mut keyboard = Vec::new();
77    for (i, opt) in options.iter().enumerate() {
78        let lower = opt.to_lowercase();
79        if lower.contains("write-in") || lower.contains("직접 입력") {
80            continue;
81        }
82        let is_checked = bitmap.chars().nth(i).unwrap_or('0') == '1';
83        let prefix = if is_checked { "✅ " } else { "⬜ " };
84        let button_text = format!("{}{}", prefix, opt);
85        let callback_data = format!("ask_mul:{}:{}:{}", sess_id, i, bitmap);
86        keyboard.push(vec![teloxide::types::InlineKeyboardButton::callback(button_text, callback_data)]);
87    }
88    keyboard.push(vec![
89        teloxide::types::InlineKeyboardButton::callback(
90            "완료 (Submit)",
91            format!("ask_sub:{}:{}", sess_id, bitmap),
92        ),
93    ]);
94    if show_prev {
95        keyboard.push(vec![
96            teloxide::types::InlineKeyboardButton::callback("⬅️ 이전 질문 (Prev)", format!("ask_prev:{}", sess_id)),
97            teloxide::types::InlineKeyboardButton::callback("⏭️ 건너뛰기 (Skip)", format!("ask_skip:{}", sess_id)),
98        ]);
99    } else {
100        keyboard.push(vec![
101            teloxide::types::InlineKeyboardButton::callback("⏭️ 건너뛰기 (Skip)", format!("ask_skip:{}", sess_id)),
102        ]);
103    }
104    teloxide::types::InlineKeyboardMarkup::new(keyboard)
105}

107pub(crate) async fn handle_ask_multi_callback(
108    bot: &teloxide::Bot,
109    msg: &Message,
110    data: &str,
111    sessions: &crate::cli::antigravity::session::SessionManager,
112) {
113    let Some(sid) = data.split(':').nth(1) else { return; };
114    let Some(idx) = data.split(':').nth(2).and_then(|i| i.parse::<usize>().ok()) else { return; };
115    let Some(bitmap) = data.split(':').nth(3) else { return; };
116    if let Some(mut state) = sessions.get_ask_state(sid).await {
117        let q = &state.questions[state.current_index];
118        if let Some(opt) = q.options.get(idx) {
119            if opt.to_lowercase().contains("write-in") || opt.contains("직접 입력") {
120                state.waiting_for_write_in = true;
121                sessions.set_ask_state(sid, state).await;
122                let txt = format!("{}\n\n✍️ **[직접 입력 대기 중...]**\n채팅창에 직접 원하는 답변을 입력한 후 전송해 주세요.", msg.text().unwrap_or(""));
123                let html = super::formatting::markdown_to_telegram_html(&txt);
124                let _ = bot.edit_message_text(msg.chat.id, msg.id, html).parse_mode(teloxide::types::ParseMode::Html).await;
125                return;
126            }
127        }
128        let nb: String = bitmap.chars().enumerate().map(|(i, c)| if i == idx { if c == '1' { '0' } else { '1' } } else { c }).collect();
129        state.current_bitmap = nb.clone();
130        sessions.set_ask_state(sid, state.clone()).await;
131        let markup = build_multi_select_keyboard(sid, &q.options, &nb, state.current_index > 0);
132        let _ = bot.edit_message_reply_markup(msg.chat.id, msg.id).reply_markup(markup).await;
133    }
134}

136pub(crate) fn extract_options_from_markup(reply_markup: &teloxide::types::InlineKeyboardMarkup) -> Vec<String> {
137    let mut options = Vec::new();
138    for row in &reply_markup.inline_keyboard {
139        for button in row {
140            if let teloxide::types::InlineKeyboardButtonKind::CallbackData(cbd) = &button.kind {
141                if cbd.starts_with("ask_mul:") {
142                    let opt_name = button.text.trim_start_matches("✅ ").trim_start_matches("⬜ ").to_string();
143                    options.push(opt_name);
144                }
145            }
146        }
147    }
148    options
149}

151fn build_prev_bitmap(prev_ans: &str, options: &[String]) -> String {
152    if prev_ans.is_empty() {
153        return "0".repeat(options.len());
154    }
155    let parts: Vec<&str> = prev_ans.split(", ").map(|s| s.trim()).collect();
156    let mut bm = String::new();
157    for opt in options {
158        if parts.contains(&opt.as_str()) {
159            bm.push('1');
160        } else {
161            bm.push('0');
162        }
163    }
164    bm
165}

167pub(crate) fn get_multiselect_keystrokes_and_options(
168    reply_markup: &teloxide::types::InlineKeyboardMarkup,
169    bitmap: &str,
170    prev_ans: &str,
171    full_options: &[String],
172) -> (String, Vec<String>) {
173    let mut selected = Vec::new();
174    let mut keystrokes = String::new();
175    let mut checked_indices = Vec::new();
176
177    let prev_bitmap = build_prev_bitmap(prev_ans, full_options);
178
179    for row in &reply_markup.inline_keyboard {
180        for button in row {
181            if let teloxide::types::InlineKeyboardButtonKind::CallbackData(cbd) = &button.kind {
182                if cbd.starts_with("ask_mul:") {
183                    if let Some(opt_index) = cbd.split(':').nth(2).and_then(|s| s.parse::<usize>().ok()) {
184                        let is_checked = bitmap.chars().nth(opt_index).unwrap_or('0') == '1';
185                        let was_checked = prev_bitmap.chars().nth(opt_index).unwrap_or('0') == '1';
186                        let opt_name = button.text.trim_start_matches("✅ ").trim_start_matches("⬜ ").to_string();
187                        if is_checked {
188                            selected.push(opt_name);
189                        }
190                        if is_checked != was_checked {
191                            checked_indices.push(opt_index);
192                        }
193                    }
194                }
195            }
196        }
197    }
198
199    checked_indices.sort_unstable();
200
201    let mut last_idx = 0;
202    for &idx in &checked_indices {
203        let diff = idx - last_idx;
204        keystrokes.push_str(&"j".repeat(diff));
205        keystrokes.push(' ');
206        last_idx = idx;
207    }
208    keystrokes.push('\r');
209    (keystrokes, selected)
210}
