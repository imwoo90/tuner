1//! # Matrix Client event Listener
2//!
3//! ## Overview
4//! Listens to Matrix events, parses incoming commands, and dispatches them to the central message bus.
5//!
6//! ## Collaboration Graph
7//! - Synchronizes room states via reqwest client.
8//! - Routes commands to [`AntigravityCli`].
9//!
10//! ## Search Tags
11//! #matrix-bot, #polling-sync, #message-ingress

40impl MatrixBot {
41    pub async fn new(
42        config: Arc<CliConfig>,
43        sessions: Arc<SessionManager>,
44        cli: Arc<AntigravityCli>,
45        id_map: Arc<MatrixIdMap>,
46        bus: Arc<MessageBus>,
47    ) -> Result<Self, String> {
48        let url = url::Url::parse(&config.matrix.homeserver).map_err(|e| e.to_string())?;
49        let client = Client::new(url).await.map_err(|e| e.to_string())?;
50        Ok(Self {
51            client,
52            config,
53            sessions,
54            cli,
55            id_map,
56            queue: Arc::new(MatrixMessageQueue::new()),
57            bus,
58        })
59    }

61    async fn setup_event_handlers(&self) {
62        self.client.add_event_handler(move |ev: StrippedRoomMemberEvent, room: Room| async move {
63            if ev.content.membership == MembershipState::Invite {
64                let _ = room.join().await;
65            }
66        });
67
68        let bot2 = self.clone();
69        self.client.add_event_handler(move |ev: OriginalSyncRoomMessageEvent, room: Room, client: Client| {
70            let bot_clone = bot2.clone();
71            async move {
72                bot_clone.handle_message(ev, room, client).await;
73            }
74        });
75    }
76
77    fn start_sync_loop(&self, store_path: &Path) {
78        let client_clone = self.client.clone();
79        let token_file = store_path.join("next_batch");
80        tokio::spawn(async move {
81            let mut sync_settings = matrix_sdk::config::SyncSettings::default()
82                .timeout(std::time::Duration::from_secs(30));
83
84            if token_file.exists() {
85                if let Ok(token) = std::fs::read_to_string(&token_file) {
86                    sync_settings = sync_settings.token(token.trim().to_string());
87                }
88            }
89
90            loop {
91                match client_clone.sync_once(sync_settings.clone()).await {
92                    Ok(response) => {
93                        let next_batch = response.next_batch;
94                        let _ = std::fs::write(&token_file, &next_batch);
95                        sync_settings = sync_settings.token(next_batch);
96                    }
97                    Err(e) => {
98                        eprintln!("Matrix Sync Error: {:?}", e);
99                        tokio::time::sleep(std::time::Duration::from_secs(5)).await;
100                    }
101                }
102            }
103        });
104    }

106    pub async fn start(&self) -> Result<(), String> {
107        let store_path = Path::new(&self.config.matrix.store_path);
108        let user_id = &self.config.matrix.user_id;
109
110        login_or_restore(
111            &self.client,
112            store_path,
113            user_id,
114            self.config.matrix.access_token.as_deref(),
115            self.config.matrix.device_id.as_deref(),
116            self.config.matrix.password.as_deref(),
117        ).await?;
118
119        self.setup_event_handlers().await;
120        self.start_sync_loop(store_path);
121
122        Ok(())
123    }

125    fn validate_message(&self, ev: &OriginalSyncRoomMessageEvent, room: &Room, client: &Client) -> Option<(String, String)> {
126        let room_id = room.room_id().to_string();
127        let self_user_id = client.user_id()?;
128
129        if ev.sender == self_user_id {
130            return None;
131        }
132
133        if self.queue.is_duplicate(ev.event_id.as_str()) {
134            return None;
135        }
136
137        let text = match &ev.content.msgtype {
138            MessageType::Text(text_content) => &text_content.body,
139            _ => return None,
140        };
141
142        if !self.config.matrix.allowed_rooms.contains(&room_id) {
143            return None;
144        }
145        if !self.config.matrix.allowed_users.contains(&ev.sender.to_string()) {
146            return None;
147        }
148
149        let is_dm = room.name().is_none() && room.canonical_alias().is_none() && room.active_members_count() <= 2;
150        let is_addressed = is_dm || text.contains(self_user_id.as_str());
151
152        if !is_addressed {
153            return None;
154        }
155
156        let prompt = text.replace(self_user_id.as_str(), "").trim().to_string();
157        Some((prompt, room_id))
158    }

160    fn process_message_task(&self, chat_id: i64, room: Room, prompt: String) -> tokio::task::JoinHandle<()> {
161        let bot_clone = self.clone();
162        tokio::spawn(async move {
163            let lock = bot_clone.bus.lock_pool().get_chat(chat_id);
164            let _guard = lock.lock().await;
165
166            let typing_guard = match MatrixTypingGuard::new(
167                bot_clone.client.clone(),
168                &room.room_id(),
169                Duration::from_secs(3),
170                Duration::from_secs(4),
171            ).await {
172                Ok(g) => Some(g),
173                Err(_) => None,
174            };
175
176            let key = crate::session::key::SessionKey::matrix(chat_id);
177            let default_model = bot_clone.config.model.as_deref().unwrap_or("antigravity-default");
178
179            if let Ok((session, _)) = bot_clone.sessions.resolve_session(&key, &bot_clone.config.provider, default_model).await {
180
181                let session_id = session.get_session_id(&bot_clone.config.provider);
182                let opt_sid = if session_id.is_empty() { None } else { Some(&session_id[..]) };
183
184                let res = bot_clone.cli.send(&prompt, opt_sid, false, bot_clone.config.working_dir.clone()).await;
185
186                drop(typing_guard);
187
188                match res {
189                    Ok(resp) => {
190                        let mut env = crate::bus::envelope::Envelope::new(crate::bus::envelope::Origin::User, chat_id);
191                        env.result_text = resp.result;
192                        env.transport = "mx".to_string();
193                        let _ = bot_clone.bus.submit(&mut env).await;
194                    }
195                    Err(e) => {
196                        let content = RoomMessageEventContent::text_plain(format!("Error: {}", e));
197                        let _ = room.send(content).await;
198                    }
199                }
200            }
201        })
202    }
203
204    async fn handle_message(&self, ev: OriginalSyncRoomMessageEvent, room: Room, client: Client) {
205        let (prompt, room_id) = match self.validate_message(&ev, &room, &client) {
206            Some(res) => res,
207            None => return,
208        };
209
210        let chat_id = self.id_map.room_to_int(&room_id);
211
212        if prompt == "/stop" || prompt == "/abort" {
213            self.queue.drain(chat_id);
214            let content = RoomMessageEventContent::text_plain("Stopped processing.");
215            let _ = room.send(content).await;
216            return;
217        }
218
219        let task = self.process_message_task(chat_id, room, prompt);
220        self.queue.track(chat_id, task);
221    }
