19pub fn get_topic_id(msg: &Message) -> Option<i64> {
20    match &msg.kind {
21        teloxide::types::MessageKind::Common(c) if c.is_topic_message => msg.thread_id.map(|t| t as i64),
22        _ => None,
23    }
24}

26pub(crate) fn strip_mention(text: &str, bot_username: Option<&str>) -> String {
27    let u = match bot_username {
28        Some(n) if !n.is_empty() => n,
29        _ => return text.to_string(),
30    };
31    let tag = format!("@{}", u.strip_prefix('@').unwrap_or(u));
32    let lower_text = text.to_lowercase();
33    let lower_tag = tag.to_lowercase();
34
35    if let Some(idx) = lower_text.find(&lower_tag) {
36        let before = &text[..idx];
37        let after = &text[idx + tag.len()..];
38        let trimmed = format!("{}{}", before, after).trim().to_string();
39        if trimmed.is_empty() { text.to_string() } else { trimmed }
40    } else {
41        text.to_string()
42    }
43}

47pub(crate) fn build_reply_prompt(message: &Message, user_text: &str) -> String {
48    let cited = if let Some(replied) = message.reply_to_message() {
49        replied.text().or(replied.caption()).map(|s| s.trim())
50    } else {
51        None
52    };
53
54    match cited {
55        None => user_text.to_string(),
56        Some("") => user_text.to_string(),
57        Some(text) => {
58            let quoted = text
59                .lines()
60                .map(|line| format!("> {}", line))
61                .collect::<Vec<String>>()
62                .join("\n");
63            format!(
64                "The user is replying to this quoted message:\n{}\n\nThe user's message:\n{}",
65                quoted, user_text
66            )
67        }
68    }
69}

71pub(crate) fn has_media(message: &Message) -> bool {
72    message.photo().is_some()
73        || message.document().is_some()
74        || message.voice().is_some()
75        || message.audio().is_some()
76        || message.video().is_some()
77        || message.video_note().is_some()
78        || message.sticker().is_some()
79}

81pub(crate) fn parse_model_directive(text: &str) -> (Option<String>, &str) {
82    let t = text.trim();
83    if let Some(r) = t.strip_prefix("@model ") {
84        let p: Vec<&str> = r.splitn(2, ' ').collect();
85        return (Some(p[0].to_string()), p.get(1).unwrap_or(&"").trim());
86    } else if t.starts_with('@') {
87        let dir = t.split_whitespace().next().unwrap_or("");
88        let m = &dir[1..];
89        if ["opus", "sonnet", "haiku", "gpt-4o", "gpt-4-turbo"].contains(&m) {
90            return (Some(m.to_string()), t[dir.len()..].trim());
91        }
92    }
93    (None, t)
94}

97pub(crate) async fn load_sessions_cache(
98    sessions: &crate::session::manager::SessionManager,
99    cache: &crate::telegram::TopicNameCache,
100) {
101    if let Ok(all) = sessions.list_all().await {
102        for s in all {
103            if let (Some(tid), Some(tname)) = (s.topic_id, s.topic_name) {
104                cache.insert(s.chat_id, tid, tname);
105            }
106        }
107    }

110pub(crate) async fn resolve_session_model(
111    msg: &Message,
112    config: &crate::config::CliConfig,
113    sessions: &crate::session::manager::SessionManager,
114) -> String {
115    let topic_id = crate::telegram::get_topic_id(msg);
116    let key = crate::session::key::SessionKey::telegram(msg.chat.id.0, topic_id);
117    let default_model = config.model.as_deref().unwrap_or("antigravity-default");
118    if let Ok((sess, _)) = sessions.resolve_session(&key, &config.provider, default_model).await {
119        sess.model
120    } else {
121        default_model.to_string()
122    }
123}

125fn get_media_file_id_and_ext(message: &Message) -> Option<(String, String)> {
126    if let Some(photo) = message.photo() {
127        if let Some(best_size) = photo.last() {
128            return Some((best_size.file.id.clone(), "jpg".to_string()));
129        }
130    } else if let Some(doc) = message.document() {
131        let ext = doc.file_name.as_ref()
132            .and_then(|name| name.split('.').last())
133            .unwrap_or("bin")
134            .to_string();
135        return Some((doc.file.id.clone(), ext));
136    } else if let Some(voice) = message.voice() {
137        return Some((voice.file.id.clone(), "ogg".to_string()));
138    } else if let Some(audio) = message.audio() {
139        return Some((audio.file.id.clone(), "mp3".to_string()));
140    } else if let Some(video) = message.video() {
141        return Some((video.file.id.clone(), "mp4".to_string()));
142    }
143    None
144}

146pub(crate) async fn download_telegram_media(
147    bot: &teloxide::Bot,
148    message: &Message,
149    dest_dir: &std::path::Path,
150) -> Result<Option<String>, String> {
151    if cfg!(test) {
152        let ext = get_media_file_id_and_ext(message)
153            .map(|(_, e)| e)
154            .unwrap_or_else(|| "jpg".to_string());
155        return Ok(Some(format!("telegram_files/mock_media_{}.{}", message.id.0, ext)));
156    }
157    let (file_id, ext) = match get_media_file_id_and_ext(message) {
158        Some(res) => res,
159        None => return Ok(None),
160    };
161
162    let file = bot.get_file(file_id).await.map_err(|e| e.to_string())?;
163    tokio::fs::create_dir_all(dest_dir).await.map_err(|e| e.to_string())?;
164
165    let filename = format!("media_{}_{}.{}", message.chat.id, message.id, ext);
166    let dest_path = dest_dir.join(&filename);
167
168    let mut dest_file = tokio::fs::File::create(&dest_path).await.map_err(|e| e.to_string())?;
169    bot.download_file(&file.path, &mut dest_file).await.map_err(|e| e.to_string())?;
170
171    Ok(Some(format!("telegram_files/{}", filename)))
172}

174pub(crate) async fn download_and_inject_media_hint(
175    bot: &teloxide::Bot,
176    message: &Message,
177    working_dir: &std::path::Path,
178    prompt: &mut String,
179) -> Result<(), String> {
180    if has_media(message) {
181        let dest_dir = working_dir.join("telegram_files");
182        match download_telegram_media(bot, message, &dest_dir).await {
183            Ok(Some(relative_path)) => {
184                let media_hint = format!(
185                    "[SYSTEM HINT] The user attached a file. You can read/view it by calling `view_file` at path: `{}`\n\n",
186                    relative_path
187                );
188                *prompt = format!("{}{}", media_hint, prompt);
189            }
190            Ok(None) => {}
191            Err(e) => return Err(e),
192        }
193    }
194    Ok(())
195}

197pub(crate) fn prepend_reply_to_media(message: &Message, media_prompt: &str) -> String {
198    let cited = if let Some(replied) = message.reply_to_message() {
199        replied.text().or(replied.caption()).map(|s| s.trim())
200    } else {
201        None
202    };
203
204    match cited {
205        None => media_prompt.to_string(),
206        Some("") => media_prompt.to_string(),
207        Some(text) => {
208            let quoted = text
209                .lines()
210                .map(|line| format!("> {}", line))
211                .collect::<Vec<String>>()
212                .join("\n");
213            
214            let base_type = if message.photo().is_some() {
215                "an image"
216            } else if message.document().is_some() {
217                "a document"
218            } else if message.voice().is_some() {
219                "a voice message"
220            } else if message.audio().is_some() {
221                "an audio file"
222            } else if message.video().is_some() {
223                "a video"
224            } else if message.video_note().is_some() {
225                "a video note"
226            } else if message.sticker().is_some() {
227                "a sticker"
228            } else {
229                "a media file"
230            };
231            let media_type = format!("{} (the attached file below).", base_type);
232
233            format!(
234                "The user is replying to this quoted message:\n{}\n\nTheir reply is {}\n\n{}",
235                quoted, media_type, media_prompt
236            )
237        }
238    }
239}

245pub(crate) async fn send_startup_notification(
246    bot: teloxide::Bot,
247    sessions: std::sync::Arc<crate::session::manager::SessionManager>,
248) {
249    println!("🤖 [tuner] send_startup_notification task started");
250    match sessions.list_all().await {
251        Ok(sess_list) => {
252            println!("🤖 [tuner] Loaded {} sessions for startup notification", sess_list.len());
253            if let Some(sess) = find_last_active_session(&sess_list) {
254                let active_lang = sess.language.clone().unwrap_or_else(|| "en".to_string());
255                crate::i18n::TASK_ACTIVE_LANG.scope(active_lang, async move {
256                    let startup_msg = crate::t!("bot.startup_complete");
257                    let mut req = bot.send_message(teloxide::types::ChatId(sess.chat_id), startup_msg);
258                    if let Some(tid) = sess.topic_id {
259                        req = req.message_thread_id(tid as i32);
260                    }
261                    match req.await {
262                        Ok(_) => println!("🤖 [tuner] Startup notification sent successfully to chat_id: {}", sess.chat_id),
263                        Err(e) => eprintln!("❌ [tuner] Failed to send startup notification to chat_id {}: {:?}", sess.chat_id, e),
264                    }
265                }).await;
266            } else {
267                println!("🤖 [tuner] No active Telegram session found for startup notification");
268            }
269        }
270        Err(e) => {
271            eprintln!("❌ [tuner] Failed to list sessions for startup notification: {:?}", e);
272        }
273    }
274}
