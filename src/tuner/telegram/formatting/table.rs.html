1//! # Markdown Table Formatter and Parser
2//!
3//! Extracts markdown pipe-tables, parses their columns and alignment rules, and formats them
4//! into fixed-width monospace layouts wrapped in pre/code blocks for clean grid presentation.
5
6//! 
7//! ## Search Tags
8//! #table
9
10use regex::Regex;
11
12fn parse_table_row(line: &str) -> Vec<String> {
13    let stripped = line.trim().trim_start_matches('|').trim_end_matches('|');
14    stripped.split('|').map(|cell| cell.trim().to_string()).collect()
15}
16
17fn is_separator_row(line: &str) -> bool {
18    let re = Regex::new(r"^\s*\|?[\s:]*-{2,}[\s:]*(\|[\s:]*-{2,}[\s:]*)*\|?\s*$").unwrap();
19    re.is_match(line)
20}
21
22fn format_table(lines: &[String]) -> String {
23    let mut rows: Vec<Vec<String>> = Vec::new();
24    for line in lines {
25        if is_separator_row(line) {
26            continue;
27        }
28        rows.push(parse_table_row(line));
29    }
30
31    if rows.is_empty() {
32        return lines.join("\n");
33    }
34
35    let n_cols = rows.iter().map(|r| r.len()).max().unwrap_or(0);
36    for row in &mut rows {
37        while row.len() < n_cols {
38            row.push(String::new());
39        }
40    }
41
42    let mut widths = vec![0; n_cols];
43    for col in 0..n_cols {
44        let mut max_w = 0;
45        for row in &rows {
46            max_w = max_w.max(row[col].chars().count());
47        }
48        widths[col] = max_w;
49    }
50
51    let mut out = Vec::new();
52    for (i, row) in rows.iter().enumerate() {
53        let mut cells = Vec::new();
54        for (c, cell) in row.iter().enumerate() {
55            let width = widths[c];
56            let padding = width.saturating_sub(cell.chars().count());
57            let padded_cell = format!("{}{}", cell, " ".repeat(padding));
58            cells.push(padded_cell);
59        }
60        out.push(cells.join("  "));
61        if i == 0 && rows.len() > 1 {
62            let divider: Vec<String> = widths.iter().map(|&w| "─".repeat(w)).collect();
63            out.push(divider.join("  "));
64        }
65    }
66
67    out.join("\n")
68}

70pub fn extract_tables(src: &str, table_blocks: &mut Vec<String>) -> String {
71    let mut out_lines = Vec::new();
72    let mut table_buf = Vec::new();
73
74    let flush = |table_buf: &mut Vec<String>, table_blocks: &mut Vec<String>, out_lines: &mut Vec<String>| {
75        if table_buf.len() >= 2 {
76            let idx = table_blocks.len();
77            table_blocks.push(format_table(table_buf));
78            out_lines.push(super::helpers::placeholder("TB", idx));
79        } else {
80            out_lines.extend(table_buf.clone());
81        }
82        table_buf.clear();
83    };
84
85    let check_table_line_re = Regex::new(r"\|.*\|").unwrap();
86    for line in src.split('\n') {
87        if line.contains('|') && check_table_line_re.is_match(line.trim()) {
88            table_buf.push(line.to_string());
89        } else {
90            if !table_buf.is_empty() {
91                let mut tb = table_buf.clone();
92                flush(&mut tb, table_blocks, &mut out_lines);
93                table_buf.clear();
94            }
95            out_lines.push(line.to_string());
96        }
97    }
98    if !table_buf.is_empty() {
99        let mut tb = table_buf.clone();
100        flush(&mut tb, table_blocks, &mut out_lines);
101    }
102
103    out_lines.join("\n")
104}
